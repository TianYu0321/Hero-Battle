# 任务卡：成就系统

## 参考来源

1. **Batocera RetroAchievements** — 成就解锁通知（左上角badge弹出）、进度追踪、成就列表UI
2. **3ddelano/epic-online-services-godot** — HAchievements接口设计（解锁/查询/进度），信号驱动
3. **论文 AGT (Achievement Tracker)** — 成就触发器绑定UI事件，进度计数器，完成时通知+弹窗

---

## 核心设计

轻量级成就系统，与现有 **SaveManager** 和 **FeedbackManager** 深度集成。

### 成就分类

| 类别 | 说明 | 示例 |
|------|------|------|
| **冒险** | 爬塔进度相关 | 到达第5层、到达第10层、首次通关 |
| **战斗** | 战斗表现 | 首次胜利、无伤通关、一击必杀 |
| **收集** | 角色/道具 | 解锁第一个伙伴、收集10个道具 |
| **累计** | 长期统计 | 累计10000金币、进行50场冒险 |
| **特殊** | 趣味挑战 | 只用主角通关、连续3场无伤 |

    for id in ACHIEVEMENTS.keys():
        var data: Dictionary = ACHIEVEMENTS[id]
        if data.get("category", -1) == category:
            result[id] = data

static func get_unlockable_achievements(include_hidden: bool = false) -> Dictionary:
    var result := {}
    for id in ACHIEVEMENTS.keys():
        var data: Dictionary = ACHIEVEMENTS[id]
        if not data.get("hidden", false) or include_hidden:
            result[id] = data
    return result

## Step 2：AchievementManager（Autoload Singleton）

```gdscript
## achievement_manager.gd — Autoload
extends Node

signal achievement_unlocked(achievement_id: String, achievement_data: Dictionary)
signal achievement_progress_updated(achievement_id: String, current: int, target: int)
signal reward_granted(gold: int)

## 运行时进度（非持久化，只记录当前RUN的临时计数）
var _session_stats: Dictionary = {
    "kill_count": 0,
    "critical_count": 0,
    "heal_amount": 0,
    "damage_dealt": 0,
    "no_damage_wins": 0,
    "current_run_no_damage": true,
}

## 从全局存档读取的解锁状态
var _unlocked_achievements: Array[String] = []
var _achievement_progress: Dictionary = {}  # 累计进度（如累计金币、累计击杀）

func _ready() -> void:
    _load_from_save()

## ========== 持久化加载/保存 ==========

func _load_from_save() -> void:
    var global: Dictionary = SaveManager.load_global()
    _unlocked_achievements = global.get("unlocked_achievements", [])
    _achievement_progress = global.get("achievement_progress", {})

func _save_to_disk() -> void:
    var global: Dictionary = SaveManager.load_global()
    global["unlocked_achievements"] = _unlocked_achievements.duplicate()
    global["achievement_progress"] = _achievement_progress.duplicate()
    SaveManager.save_global(global)

## ========== 核心API：检查并解锁 ==========

func check_and_unlock(condition_type: AchievementData.ConditionType, value: int = 1, context: Dictionary = {}) -> void:
    ## 遍历所有成就，检查是否有满足条件的未解锁成就
    for id in AchievementData.ACHIEVEMENTS.keys():
        if _is_unlocked(id):
            continue  # 已解锁，跳过
        
        var data: Dictionary = AchievementData.ACHIEVEMENTS[id]
        if data.get("condition_type", -1) != condition_type:
            continue  # 条件类型不匹配
        
        var target: int = data.get("target_value", 1)
        var current: int = _get_current_progress(id, data, context)
        
        if current >= target:
            _unlock_achievement(id)

## ========== 特定事件触发 ==========

func on_battle_won(battle_result: Dictionary) -> void:
    ## 战斗胜利
    check_and_unlock(AchievementData.ConditionType.WIN_BATTLE, 1)
    
    ## 检查无伤
    var hero_damage_taken: int = battle_result.get("hero_damage_taken", 0)
    if hero_damage_taken <= 0:
        _session_stats["no_damage_wins"] += 1
        check_and_unlock(AchievementData.ConditionType.NO_DAMAGE, _session_stats["no_damage_wins"])
    
    ## 累计击杀
    var enemies_killed: int = battle_result.get("enemies_killed", 1)
    _add_progress(AchievementData.ConditionType.KILL_COUNT, enemies_killed)
    
    ## 暴击数
    var crit_count: int = battle_result.get("critical_count", 0)
    if crit_count > 0:
        _add_progress(AchievementData.ConditionType.CRITICAL_COUNT, crit_count)
    
    ## 伤害统计
    var damage_dealt: int = battle_result.get("damage_dealt", 0)
    if damage_dealt > 0:
        _add_progress(AchievementData.ConditionType.DAMAGE_DEALT, damage_dealt)

func on_floor_reached(floor_num: int) -> void:
    ## 到达层数
    check_and_unlock(AchievementData.ConditionType.REACH_FLOOR, floor_num, {"floor": floor_num})
    
    ## 检查无伤通关（整场RUN无伤）
    if _session_stats["current_run_no_damage"]:
        ## 这个需要通关时检查，不是在每层
        pass

func on_run_started(hero_id: String, partner_ids: Array[String]) -> void:
    ## 重置当前RUN统计
    _session_stats["kill_count"] = 0
    _session_stats["critical_count"] = 0
    _session_stats["heal_amount"] = 0
    _session_stats["damage_dealt"] = 0
    _session_stats["no_damage_wins"] = 0
    _session_stats["current_run_no_damage"] = true
    
    ## 检查首次冒险
    check_and_unlock(AchievementData.ConditionType.RUN_COUNT, _get_total_runs())
    
    ## 检查孤狼（无伙伴）
    if partner_ids.is_empty():
        ## 特殊条件，标记待检查
        _session_stats["lone_wolf_active"] = true

func on_run_ended(victory: bool, floor_reached: int, partner_ids: Array[String]) -> void:
    if victory:
        ## 孤狼检查
        if _session_stats.get("lone_wolf_active", false) and partner_ids.is_empty():
            check_and_unlock(AchievementData.ConditionType.REACH_FLOOR, floor_reached, {"lone_wolf": true})
    
    ## 重置
    _session_stats["lone_wolf_active"] = false

func on_partner_unlocked(partner_id: String) -> void:
    ## 累计伙伴数
    var total_partners: int = _get_unlocked_partner_count()
    check_and_unlock(AchievementData.ConditionType.UNLOCK_PARTNER, total_partners)

func on_gold_earned(amount: int) -> void:
    ## 累计金币
    _add_progress(AchievementData.ConditionType.GOLD_TOTAL, amount)

func on_heal(amount: int) -> void:
    _session_stats["heal_amount"] += amount
    _add_progress(AchievementData.ConditionType.HEAL_AMOUNT, amount)

func on_hero_damage_taken(amount: int) -> void:
    if amount > 0:
        _session_stats["current_run_no_damage"] = false

## ========== 内部进度管理 ==========

func _get_current_progress(achievement_id: String, data: Dictionary, context: Dictionary) -> int:
    var condition_type: int = data.get("condition_type", -1)
    
    match condition_type:
        AchievementData.ConditionType.REACH_FLOOR:
            return context.get("floor", 0)
        AchievementData.ConditionType.WIN_BATTLE:
            return _session_stats.get("win_count", 0) + 1
        AchievementData.ConditionType.NO_DAMAGE:
            return _session_stats.get("no_damage_wins", 0)
        AchievementData.ConditionType.KILL_COUNT:
            return _get_progress(AchievementData.ConditionType.KILL_COUNT)
        AchievementData.ConditionType.CRITICAL_COUNT:
            return _get_progress(AchievementData.ConditionType.CRITICAL_COUNT)
        AchievementData.ConditionType.GOLD_TOTAL:
            return _get_progress(AchievementData.ConditionType.GOLD_TOTAL)
        AchievementData.ConditionType.RUN_COUNT:
            return _get_total_runs()
        AchievementData.ConditionType.UNLOCK_PARTNER:
            return _get_unlocked_partner_count()
        AchievementData.ConditionType.HEAL_AMOUNT:
            return _get_progress(AchievementData.ConditionType.HEAL_AMOUNT)
        AchievementData.ConditionType.DAMAGE_DEALT:
            return _get_progress(AchievementData.ConditionType.DAMAGE_DEALT)
        _:
            return 0

func _add_progress(condition_type: AchievementData.ConditionType, amount: int) -> void:
    var key := str(condition_type)
    var current: int = _achievement_progress.get(key, 0)
    _achievement_progress[key] = current + amount
    
    ## 检查所有该类型的成就
    check_and_unlock(condition_type, current + amount)

func _get_progress(condition_type: AchievementData.ConditionType) -> int:
    return _achievement_progress.get(str(condition_type), 0)

func _get_total_runs() -> int:
    var global: Dictionary = SaveManager.load_global()
    return global.get("total_runs", 0)

func _get_unlocked_partner_count() -> int:
    var global: Dictionary = SaveManager.load_global()
    return global.get("unlocked_partners", []).size()

## ========== 解锁流程 ==========

func _unlock_achievement(achievement_id: String) -> void:
    if _is_unlocked(achievement_id):
        return
    
    _unlocked_achievements.append(achievement_id)
    
    var data: Dictionary = AchievementData.get_achievement(achievement_id)
    
    ## 发放奖励
    var reward: int = data.get("reward_gold", 0)
    if reward > 0:
        GameManager.total_gold += reward  # 或加到全局存档
        reward_granted.emit(reward)
    
    ## 保存
    _save_to_disk()
    
    ## 发射信号
    achievement_unlocked.emit(achievement_id, data)
    
    ## 显示解锁通知
    _show_unlock_notification(achievement_id, data)

func _is_unlocked(achievement_id: String) -> bool:
    return _unlocked_achievements.has(achievement_id)

## ========== 解锁通知UI ==========

func _show_unlock_notification(achievement_id: String, data: Dictionary) -> void:
    var notification := preload("res://scenes/ui/achievement_unlock_notification.tscn").instantiate()
    notification.setup(data)
    
    ## 添加到最顶层
    var canvas := CanvasLayer.new()
    canvas.layer = 90  # 低于TransitionManager(100)，高于普通UI(5)
    canvas.name = "AchievementNotification"
    get_tree().root.add_child(canvas)
    canvas.add_child(notification)

## ========== 查询API ==========

func get_unlocked_ids() -> Array[String]:
    return _unlocked_achievements.duplicate()

func get_unlock_count() -> int:
    return _unlocked_achievements.size()

func get_total_count() -> int:
    return AchievementData.ACHIEVEMENTS.size()

func is_unlocked(achievement_id: String) -> bool:
    return _is_unlocked(achievement_id)

func get_progress_percent(achievement_id: String) -> float:
    var data: Dictionary = AchievementData.get_achievement(achievement_id)
    if data.is_empty():
        return 0.0
    
    var target: int = data.get("target_value", 1)
    var current: int = _get_current_progress(achievement_id, data, {})
    return clampf(float(current) / float(target), 0.0, 1.0)

## ========== 调试 ==========

func force_unlock(achievement_id: String) -> void:
    ## 调试用：强制解锁
    _unlock_achievement(achievement_id)

func reset_all() -> void:
    ## 调试用：重置所有成就
    _unlocked_achievements.clear()
    _achievement_progress.clear()
    _save_to_disk()
```

## achievement_unlock_notification.gd
extends PanelContainer

const SLIDE_DURATION := 0.5
const DISPLAY_DURATION := 3.0
const SLIDE_OUT_DURATION := 0.4

func setup(data: Dictionary) -> void:
    var name_label: Label = $VBoxContainer/NameLabel
    var desc_label: Label = $VBoxContainer/DescLabel
    var icon_tex: TextureRect = $VBoxContainer/HBoxContainer/IconTexture
    var reward_label: Label = $VBoxContainer/HBoxContainer/RewardLabel
    
    name_label.text = data.get("name", "???")
    desc_label.text = data.get("description", "")
    
    ## 图标
    var icon_path: String = data.get("icon", "")
    if not icon_path.is_empty():
        var tex: Texture2D = load(icon_path)
        if tex != null:
            icon_tex.texture = tex
    
    ## 奖励
    var reward: int = data.get("reward_gold", 0)
    if reward > 0:
        reward_label.text = "+%d金币" % reward
        reward_label.add_theme_color_override("font_color", Color(0.85, 0.65, 0.15, 1))
    else:
        reward_label.visible = false
    
    ## 样式：金色边框成就感
    var style := StyleBoxFlat.new()
    style.bg_color = Color(0.98, 0.96, 0.88, 1.0)  # 淡金黄底
    style.border_color = Color(0.85, 0.65, 0.15, 1.0)  # 金色边框
    style.border_width_left = 3
    style.border_width_top = 3
    style.border_width_right = 3
    style.border_width_bottom = 3
    style.corner_radius_top_left = 12
    style.corner_radius_top_right = 12
    style.corner_radius_bottom_left = 12
    style.corner_radius_bottom_right = 12
    style.shadow_color = Color(0.85, 0.65, 0.15, 0.25)
    style.shadow_size = 12
    style.shadow_offset = Vector2(0, 4)
    add_theme_stylebox_override("panel", style)
    
    ## 尺寸和初始位置
    custom_minimum_size = Vector2(360, 100)
    position = Vector2(1280, 80)  # 右侧屏幕外
    
    ## 滑入动画
    _animate_in()

func _animate_in() -> void:
    var tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT)
    tween.tween_property(self, "position:x", 880, SLIDE_DURATION)  # 滑到 x=880
    
    ## 伴随缩放弹跳
    scale = Vector2(0.8, 0.8)
    tween.parallel().tween_property(self, "scale", Vector2.ONE, SLIDE_DURATION)
    
    ## 伴随闪光效果
    modulate = Color(1.5, 1.5, 1.0, 1.0)  # 闪光
    tween.parallel().tween_property(self, "modulate", Color.WHITE, 0.3)
    
    await tween.finished
    
    ## 等待显示时间
    await get_tree().create_timer(DISPLAY_DURATION).timeout
    
    ## 滑出
    _animate_out()

func _animate_out() -> void:
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_IN)
    tween.tween_property(self, "position:x", 1280 + size.x, SLIDE_OUT_DURATION)
    tween.parallel().tween_property(self, "modulate:a", 0.0, SLIDE_OUT_DURATION)
    
    await tween.finished
    queue_free()
```

        scroll.name = AchievementData.get_category_name(cat)
        tab_container.add_child(scroll)
        
        var grid := GridContainer.new()
        grid.columns = 4
        grid.add_theme_constant_override("h_separation", 12)
        grid.add_theme_constant_override("v_separation", 12)
        scroll.add_child(grid)
        
        var achievements: Dictionary = AchievementData.get_achievements_by_category(cat)
        for id in achievements.keys():
            var card := _create_achievement_card(id, achievements[id])
            grid.add_child(card)

func _create_achievement_card(id: String, data: Dictionary) -> PanelContainer:
    var card := PanelContainer.new()
    card.custom_minimum_size = Vector2(200, 120)
    
    var is_unlocked: bool = AchievementManager.is_unlocked(id)
    var is_hidden: bool = data.get("hidden", false) and not is_unlocked
    
    ## 样式
    var style := StyleBoxFlat.new()
    if is_unlocked:
        style.bg_color = Color(1.0, 0.98, 0.92, 1.0)  # 淡金底
        style.border_color = Color(0.85, 0.65, 0.15, 1.0)
    elif is_hidden:
        style.bg_color = Color(0.92, 0.92, 0.94, 1.0)  # 灰色（隐藏）
        style.border_color = Color(0.75, 0.75, 0.78, 1.0)
    else:
        style.bg_color = Color(1.0, 1.0, 1.0, 1.0)  # 白底
        style.border_color = Color(0.85, 0.85, 0.85, 1.0)
    
    style.border_width_left = 2
    style.border_width_top = 2
    style.border_width_right = 2
    style.border_width_bottom = 2
    style.corner_radius_top_left = 10
    style.corner_radius_top_right = 10
    style.corner_radius_bottom_left = 10
    style.corner_radius_bottom_right = 10
    card.add_theme_stylebox_override("panel", style)
    
    var vbox := VBoxContainer.new()
    vbox.alignment = BoxContainer.ALIGNMENT_CENTER
    vbox.add_theme_constant_override("separation", 4)
    card.add_child(vbox)
    
    ## 图标
    var icon := TextureRect.new()
    icon.custom_minimum_size = Vector2(48, 48)
    icon.expand_mode = TextureRect.EXPAND_FIT_WIDTH_PROPORTIONAL
    icon.stretch_mode = TextureRect.STRETCH_KEEP_ASPECT_CENTERED
    
    if is_hidden:
        icon.modulate = Color(0.5, 0.5, 0.55, 0.3)  # 剪影
    elif is_unlocked:
        var icon_path: String = data.get("icon", "")
        if not icon_path.is_empty():
            var tex: Texture2D = load(icon_path)
            if tex != null:
                icon.texture = tex
    else:
        ## 未解锁：灰度
        var icon_path: String = data.get("icon", "")
        if not icon_path.is_empty():
            var tex: Texture2D = load(icon_path)
            if tex != null:
                icon.texture = tex
        icon.modulate = Color(0.5, 0.5, 0.55, 0.5)  # 半透灰
    
    vbox.add_child(icon)
    
    ## 名称
    var name_label := Label.new()
    if is_hidden:
        name_label.text = "???"
        name_label.add_theme_color_override("font_color", Color(0.5, 0.5, 0.55, 1))
    else:
        name_label.text = data.get("name", "???")
        if is_unlocked:
            name_label.add_theme_color_override("font_color", Color(0.2, 0.2, 0.2, 1))
        else:
            name_label.add_theme_color_override("font_color", Color(0.6, 0.6, 0.65, 1))
    name_label.add_theme_font_size_override("font_size", 14)
    name_label.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
    vbox.add_child(name_label)
    
    ## 进度条（未解锁的显示进度）
    if not is_unlocked and not is_hidden:
        var progress: float = AchievementManager.get_progress_percent(id)
        var progress_bar := ProgressBar.new()
        progress_bar.custom_minimum_size = Vector2(160, 6)
        progress_bar.max_value = 1.0
        progress_bar.value = progress
        progress_bar.show_percentage = false
        
        var fg := StyleBoxFlat.new()
        fg.bg_color = Color(0.85, 0.65, 0.15, 1.0)
        fg.corner_radius_top_left = 3
        fg.corner_radius_top_right = 3
        fg.corner_radius_bottom_left = 3
        fg.corner_radius_bottom_right = 3
        progress_bar.add_theme_stylebox_override("fill", fg)
        
        var bg := StyleBoxFlat.new()
        bg.bg_color = Color(0.9, 0.9, 0.92, 1.0)
        bg.corner_radius_top_left = 3
        bg.corner_radius_top_right = 3
        bg.corner_radius_bottom_left = 3
        bg.corner_radius_bottom_right = 3
        progress_bar.add_theme_stylebox_override("background", bg)
        
        vbox.add_child(progress_bar)
    
    return card
```
