# 任务卡：精灵图路径统一配置

## 说明

当前项目中角色/怪物头像、立绘、背景图的加载路径散落在各场景脚本中硬编码，导致：
- 更换资源目录结构时代码多处需要修改
- 拼写错误难以发现
- 不同场景引用同一资源路径不一致
- 缺失资源时崩溃或显示异常

本任务将所有图片资源路径集中到一个配置文件中管理。

## Step 2：ResourcePaths Autoload

```gdscript
## resource_paths.gd — Autoload Singleton
extends Node

class_name ResourcePaths

## ========== 基础路径常量 ==========
const BASE_PATH := "res://assets/"

## ========== 角色头像（64x64）==========

static func get_hero_avatar(hero_id: String) -> String:
    var path := BASE_PATH + "characters/heroes/%s/avatar.png" % hero_id
    if not _file_exists(path):
        push_warning("Hero avatar not found: %s" % path)
        return _get_fallback_avatar()
    return path

static func get_partner_avatar(partner_id: String) -> String:
    var path := BASE_PATH + "characters/partners/%s/avatar.png" % partner_id
    if not _file_exists(path):
        push_warning("Partner avatar not found: %s" % path)
        return _get_fallback_avatar()
    return path

static func get_enemy_avatar(enemy_id: String) -> String:
    var path := BASE_PATH + "characters/enemies/%s/avatar.png" % enemy_id
    if not _file_exists(path):
        push_warning("Enemy avatar not found: %s" % path)
        return _get_fallback_enemy_avatar()
    return path

## ========== 角色立绘（512x512）==========

static func get_hero_portrait(hero_id: String) -> String:
    var path := BASE_PATH + "characters/heroes/%s/portrait.png" % hero_id
    if not _file_exists(path):
        push_warning("Hero portrait not found: %s" % path)
        return _get_fallback_portrait()
    return path

static func get_partner_portrait(partner_id: String) -> String:
    var path := BASE_PATH + "characters/partners/%s/portrait.png" % partner_id
    if not _file_exists(path):
        push_warning("Partner portrait not found: %s" % path)
        return _get_fallback_portrait()
    return path

static func get_enemy_portrait(enemy_id: String) -> String:
    var path := BASE_PATH + "characters/enemies/%s/portrait.png" % enemy_id
    if not _file_exists(path):
        push_warning("Enemy portrait not found: %s" % path)
        return _get_fallback_enemy_portrait()
    return path

## ========== 背景图 ==========

static func get_menu_background() -> String:
    return BASE_PATH + "backgrounds/menu/main_menu_bg.jpg"

static func get_hero_select_background() -> String:
    return BASE_PATH + "backgrounds/menu/hero_select_bg.jpg"

static func get_battle_background(battle_type: String = "normal") -> String:
    match battle_type:
        "elite": return BASE_PATH + "backgrounds/battle/elite_battle_bg.jpg"
        "boss": return BASE_PATH + "backgrounds/battle/boss_battle_bg.jpg"
        _: return BASE_PATH + "backgrounds/battle/normal_battle_bg.jpg"

static func get_tower_background() -> String:
    return BASE_PATH + "backgrounds/run/tower_bg.jpg"

## ========== UI图标 ==========

static func get_icon(icon_name: String) -> String:
    var path := BASE_PATH + "ui/icons/%s.png" % icon_name
    if not _file_exists(path):
        push_warning("UI icon not found: %s" % path)
        return ""
    return path

## ========== 粒子纹理 ==========

static func get_particle_texture(particle_name: String) -> String:
    var path := BASE_PATH + "ui/particles/%s.png" % particle_name
    if not _file_exists(path):
        push_warning("Particle texture not found: %s" % path)
        return ""
    return path

## ========== 特效 ==========

static func get_effect_texture(effect_name: String) -> String:
    var path := BASE_PATH + "effects/%s.png" % effect_name
    if not _file_exists(path):
        push_warning("Effect texture not found: %s" % path)
        return ""
    return path

## ========== 兜底资源 ==========

static func _get_fallback_avatar() -> String:
    return BASE_PATH + "ui/icons/unknown_avatar.png"

static func _get_fallback_enemy_avatar() -> String:
    return BASE_PATH + "ui/icons/unknown_enemy.png"

static func _get_fallback_portrait() -> String:
    return BASE_PATH + "ui/icons/unknown_portrait.png"

static func _get_fallback_enemy_portrait() -> String:
    return BASE_PATH + "ui/icons/unknown_enemy_portrait.png"

## ========== 辅助函数 ==========

static func _file_exists(path: String) -> bool:
    return FileAccess.file_exists(path)

static func load_texture_safe(path: String) -> Texture2D:
    if path.is_empty():
        return null
    var tex: Texture2D = load(path)
    if tex == null:
        push_warning("Failed to load texture: %s" % path)
    return tex
```

## Step 5：缺失资源兜底

```gdscript
# 在 ResourcePaths 中添加：当资源缺失时的处理

static func get_hero_avatar_with_fallback(hero_id: String) -> Texture2D:
    var tex := load_texture_safe(get_hero_avatar(hero_id))
    if tex == null:
        # 生成纯色占位图
        return _generate_color_placeholder(Color.GRAY, 64, 64)
    return tex

static func _generate_color_placeholder(color: Color, width: int, height: int) -> Texture2D:
    var img := Image.create(width, height, false, Image.FORMAT_RGBA8)
    img.fill(color)
    
    # 添加边框
    for x in range(width):
        img.set_pixel(x, 0, Color.WHITE)
        img.set_pixel(x, height - 1, Color.WHITE)
    for y in range(height):
        img.set_pixel(0, y, Color.WHITE)
        img.set_pixel(width - 1, y, Color.WHITE)
    
    var tex := ImageTexture.create_from_image(img)
    return tex
```

## Step 6：资源加载缓存（可选优化）

```gdscript
# 如果需要频繁加载同一纹理（如战斗中的多次刷新），添加缓存：

var _texture_cache: Dictionary = {}

static func load_texture_cached(path: String) -> Texture2D:
    if path in _texture_cache:
        return _texture_cache[path]
    
    var tex := load_texture_safe(path)
    if tex != null:
        _texture_cache[path] = tex
    return tex

static func clear_cache() -> void:
    _texture_cache.clear()
```
