# 任务卡：场景过渡动画衔接统一

## 参考来源（已分析的真实项目）

1. **Pandita-Studio/scene-switcher-godot-addon** — Godot 4场景切换插件
   - `SceneSwitcher` autoload singleton
   - 动态创建 `CanvasLayer(layer=100) + ColorRect` 全屏遮罩
   - API：`SceneSwitcher.switch_to(path)`，支持 `await SceneSwitcher.finished`
   - 启动时自动fade out，切换时fade in → change_scene → wait → fade out

2. **glass-brick/Scene-Manager** — Godot 4专业场景管理器
   - `change_scene()` / `fade_out()` / `fade_in()` / `fade_in_place()`
   - 可配置：`speed`, `color`, `pattern`, `wait_time`, `ease`, `skip_fade_in/out`
   - 信号：`scene_unloaded`, `scene_loaded`, `fade_complete`, `transition_finished`
   - `is_transitioning` bool防止并发切换

3. **murikistudio/simple-gui-transitions** — UI过渡插件
   - 支持 `Fade` / `Slide(left/right/up/down)` / `Scale(vertical/horizontal/both)`
   - 动画进入/离开分别配置
   - `Duration` + `Delay`（各元素错开延迟）

4. **jhlothamer/save_and_load_demo** — Transition Manager addon
   - fade-out/fade-in效果封装为autoload
   - 场景切换前保存状态，切换后恢复

5. **ectucker1/EctGodotUtils** — `TransitionLayer` autoload
   - 配合 `CameraEffects` 一起使用

## 核心设计

用一个 **TransitionManager** autoload 统一管理所有场景切换，消除各场景自行写淡入淡出的混乱。

### 过渡类型配置

| 切换路径 | 类型 | 颜色 | 时长 | 说明 |
|----------|------|------|------|------|
| 启动→主菜单 | FadeOut | 黑色 | 0.6s | 从黑屏淡入 |
| 主菜单→选人 | Fade | 米白色 #F5F0E8 | 0.4s | 温暖过渡 |
| 选人→酒馆集结 | SlideLeft | 米白色 | 0.35s | 左滑进入（新场景从右向左覆盖） |
| 酒馆→爬塔主场景 | Fade | 浅蓝白 #E8EEF5 | 0.5s | 进入冒险 |
| 爬塔→战斗 | ZoomIn | 暗红 #1A0505 | 0.4s | 战斗紧张感（从中心放大进入） |
| 战斗→爬塔（胜利）| Fade | 浅蓝白 | 0.4s | 回到爬塔 |
| 战斗→游戏结束 | Fade | 暗灰 #1A1A1A | 0.8s | 失败沉重感 |
| 爬塔→商店/训练 | ScaleDown | 白色 | 0.3s | 弹窗感（场景缩小后放大） |
| 任意→暂停菜单 | FadeInPlace | 半透明黑 | 0.25s | 原地淡入（不切换场景） |

## Step 1：TransitionManager Autoload

### 节点结构

```
TransitionManager (Node, autoload)
├── CanvasLayer (layer=100, 最顶层)
│   └── Overlay (ColorRect, full rect, 初始透明)
├── PatternOverlay (TextureRect, 可选遮罩图案, 初始透明)
└── AnimationPlayer (可选, 复杂过渡用)
```

### 核心代码

```gdscript
## transition_manager.gd — Autoload Singleton
extends Node

signal transition_started(transition_name: String)
signal fade_complete()           # 黑屏完成（适合在中间做操作）
signal scene_loaded()            # 新场景加载完成
signal transition_finished()     # 整个过渡完成

enum TransitionType {
    FADE,           # 淡入淡出（通用）
    FADE_OUT,       # 只淡出（用于启动）
    FADE_IN_PLACE,  # 原地淡入淡出（不切换场景，如暂停菜单）
    SLIDE_LEFT,     # 左滑（新场景从右进入）

## ========== 核心API ==========

## 切换场景（最常用）
func switch_scene(scene_path: String, transition_key: String = "fade") -> void:
    if _is_transitioning:
        push_warning("Transition already in progress, ignoring request")
        return
    
    var config: Dictionary = TRANSITION_CONFIG.get(transition_key, TRANSITION_CONFIG["fade"])
    _execute_transition(scene_path, config)

## 原地淡入淡出（不切换场景，如暂停菜单打开/关闭）
func fade_in_place(fade_in: bool = true, transition_key: String = "pause_menu") -> void:
    if _is_transitioning:
        return
    
    var config: Dictionary = TRANSITION_CONFIG.get(transition_key, TRANSITION_CONFIG["pause_menu"])
    _execute_fade_in_place(fade_in, config)

## 等待过渡完成（协程用）
func await_transition() -> void:
    if not _is_transitioning:
        return
    await transition_finished

## ========== 过渡执行 ==========

func _execute_transition(scene_path: String, config: Dictionary) -> void:
    _is_transitioning = true
    transition_started.emit(config.get("_name", "unknown"))
    
    var trans_type: TransitionType = config.get("type", TransitionType.FADE)
    var color: Color = config.get("color", Color.BLACK)
    var duration: float = config.get("duration", DEFAULT_FADE_DURATION)
    var wait_time: float = config.get("wait_time", DEFAULT_WAIT_TIME)
    var ease_type: int = config.get("ease", DEFAULT_EASE)
    var is_reverse: bool = config.get("reverse", false)
    
    _canvas_layer.visible = true
    
    match trans_type:
        TransitionType.FADE, TransitionType.FADE_OUT:
            await _do_fade_transition(scene_path, color, duration, wait_time, ease_type, is_reverse)
        TransitionType.SLIDE_LEFT, TransitionType.SLIDE_RIGHT, TransitionType.SLIDE_UP, TransitionType.SLIDE_DOWN:
            await _do_slide_transition(scene_path, trans_type, color, duration, wait_time, ease_type)
        TransitionType.ZOOM_IN, TransitionType.ZOOM_OUT:
            await _do_zoom_transition(scene_path, trans_type, color, duration, wait_time, ease_type)
        TransitionType.SCALE_DOWN:
            await _do_scale_transition(scene_path, color, duration, wait_time, ease_type, is_reverse)
        _:
            await _do_fade_transition(scene_path, color, duration, wait_time, ease_type, false)
    
    _is_transitioning = false
    transition_finished.emit()

## --- Fade过渡（参考Pandita-Studio SceneSwitcher）---
func _do_fade_transition(scene_path: String, color: Color, duration: float, wait: float, ease_type: int, reverse: bool) -> void:
    # Phase 1: 淡入（变暗/变色）
    var target_alpha := 1.0 if not reverse else 0.0
    var start_alpha := 0.0 if not reverse else 1.0
    
    _overlay.color = Color(color.r, color.g, color.b, start_alpha)
    
    var tween_in := create_tween().set_trans(DEFAULT_TRANS).set_ease(ease_type)
    tween_in.tween_property(_overlay, "color:a", target_alpha, duration)
    await tween_in.finished
    
    # Phase 2: 切换场景（在完全黑屏/色屏时）
    if scene_path != "" and not reverse:
        var err := get_tree().change_scene_to_file(scene_path)
        if err != OK:
            push_error("Failed to change scene to: %s" % scene_path)
        scene_loaded.emit()
    
    # 等待
    if wait > 0:
        await get_tree().create_timer(wait).timeout
    
    # Phase 3: 淡出（恢复）
    var tween_out := create_tween().set_trans(DEFAULT_TRANS).set_ease(Tween.EASE_OUT)
    tween_out.tween_property(_overlay, "color:a", start_alpha, duration)
    await tween_out.finished
    
    if reverse or scene_path == "":
        _canvas_layer.visible = false

## --- Slide滑动过渡 ---
func _do_slide_transition(scene_path: String, direction: TransitionType, color: Color, duration: float, wait: float, ease_type: int) -> void:
    var slide_offset := Vector2.ZERO
    match direction:
        TransitionType.SLIDE_LEFT:
            slide_offset = Vector2(-get_viewport().get_visible_rect().size.x, 0)
        TransitionType.SLIDE_RIGHT:
            slide_offset = Vector2(get_viewport().get_visible_rect().size.x, 0)
        TransitionType.SLIDE_UP:
            slide_offset = Vector2(0, -get_viewport().get_visible_rect().size.y)
        TransitionType.SLIDE_DOWN:
            slide_offset = Vector2(0, get_viewport().get_visible_rect().size.y)
    
    # 起始位置
    _overlay.position = -slide_offset
    _overlay.color = Color(color.r, color.g, color.b, 1.0)
    
    # 滑入覆盖
    var tween_in := create_tween().set_trans(DEFAULT_TRANS).set_ease(ease_type)
    tween_in.tween_property(_overlay, "position", Vector2.ZERO, duration)
    await tween_in.finished
    
    # 切换场景
    if scene_path != "":
        get_tree().change_scene_to_file(scene_path)
        scene_loaded.emit()
    
    if wait > 0:
        await get_tree().create_timer(wait).timeout
    
    # 滑出（继续向原方向滑动直到离开屏幕）
    var tween_out := create_tween().set_trans(DEFAULT_TRANS).set_ease(Tween.EASE_OUT)
    tween_out.tween_property(_overlay, "position", slide_offset, duration)
    await tween_out.finished
    
    _canvas_layer.visible = false

## --- Zoom缩放过渡（中心放大/缩小）---
func _do_zoom_transition(scene_path: String, direction: TransitionType, color: Color, duration: float, wait: float, ease_type: int) -> void:
    var start_scale := 0.0 if direction == TransitionType.ZOOM_IN else 3.0
    var end_scale := 3.0 if direction == TransitionType.ZOOM_IN else 0.0
    
    _overlay.color = Color(color.r, color.g, color.b, 1.0)
    _overlay.pivot_offset = _overlay.size / 2
    _overlay.scale = Vector2(start_scale, start_scale)
    
    # 放大/缩小覆盖
    var tween_in := create_tween().set_trans(DEFAULT_TRANS).set_ease(ease_type)
    tween_in.tween_property(_overlay, "scale", Vector2(end_scale, end_scale), duration)
    await tween_in.finished
    
    # 切换场景
    if scene_path != "":
        get_tree().change_scene_to_file(scene_path)
        scene_loaded.emit()
    
    if wait > 0:
        await get_tree().create_timer(wait).timeout
    
    # 恢复
    var tween_out := create_tween().set_trans(DEFAULT_TRANS).set_ease(Tween.EASE_OUT)
    tween_out.tween_property(_overlay, "scale", Vector2.ONE, duration)
    tween_out.parallel().tween_property(_overlay, "color:a", 0.0, duration)
    await tween_out.finished
    
    _canvas_layer.visible = false
    _overlay.scale = Vector2.ONE

## --- Scale弹窗过渡（参考murikistudio simple-gui-transitions）---
func _do_scale_transition(scene_path: String, color: Color, duration: float, wait: float, ease_type: int, reverse: bool) -> void:
    var start_scale := 0.8 if not reverse else 1.0
    var end_scale := 1.0 if not reverse else 0.8
    var start_alpha := 0.0 if not reverse else 1.0
    var end_alpha := 1.0 if not reverse else 0.0
    
    _overlay.color = Color(color.r, color.g, color.b, start_alpha)
    _overlay.pivot_offset = _overlay.size / 2
    _overlay.scale = Vector2(start_scale, start_scale)
    
    var tween := create_tween().set_trans(DEFAULT_TRANS).set_ease(ease_type)
    tween.tween_property(_overlay, "scale", Vector2(end_scale, end_scale), duration)
    tween.parallel().tween_property(_overlay, "color:a", end_alpha, duration)
    await tween.finished
    
    if not reverse:
        if wait > 0:
            await get_tree().create_timer(wait).timeout
        # 不切换场景，保持遮罩
    else:
        _canvas_layer.visible = false
        _overlay.scale = Vector2.ONE

## --- 原地淡入淡出 ---
func _execute_fade_in_place(fade_in: bool, config: Dictionary) -> void:
    _is_transitioning = true
    
    var color: Color = config.get("color", Color(0, 0, 0, 0.6))
    var duration: float = config.get("duration", 0.25)
    var ease_type: int = config.get("ease", Tween.EASE_OUT)
    
    _canvas_layer.visible = true
    _overlay.color = Color(color.r, color.g, color.b, 0.0 if fade_in else color.a)
    
    var tween := create_tween().set_trans(DEFAULT_TRANS).set_ease(ease_type)
    tween.tween_property(_overlay, "color:a", color.a if fade_in else 0.0, duration)
    await tween.finished
    
    _is_transitioning = false
    transition_finished.emit()
    
    if not fade_in:
        _canvas_layer.visible = false
```

```gdscript
func _close_shop_popup() -> void:
    var popup: Control = $PopupLayer/ShopPopup
    
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_IN)
    tween.tween_property(popup, "scale", Vector2(0.8, 0.8), 0.2)
    tween.parallel().tween_property(popup, "modulate:a", 0.0, 0.2)
    await tween.finished
    
    popup.visible = false
    TransitionManager.fade_in_place(false, "popup_to_run")
```
