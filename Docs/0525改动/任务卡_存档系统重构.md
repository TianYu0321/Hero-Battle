# 任务卡：存档系统重构

## 参考来源（已分析的真实项目）

### 1. hi-godot/save-system-godot-claude（Godot 4完整存档库）
- **Decoupled library**：`save_system/` 零依赖，任何加入 `saveable` group 且实现 `save_data()` / `load_data()` 的节点自动参与
- **Atomic writes**：写入 `<slot>.save.tmp` → `rename()` 到正式文件，崩溃不会留下半写文件
- **Encrypted**：`FileAccess.open_encrypted_with_pass`，JSON payload 加密
- **Metadata sidecar**：每个槽位 `.save`（加密）+ `.meta.tres`（明文 Godot Resource），槽位列表无需解密
- **Schema versioning**：顶层 `schema_version` 支持迁移
- **Signals**：`saved`, `loaded`, `load_started`, `save_failed`, `load_failed`
- **HMAC-SHA256**：完整性签名，防止篡改

### 2. jhlothamer/save_and_load_demo（Godot保存加载Demo）
- **GameStateService**：autoload singleton，监听场景切换自动保存/加载
- **GameStateHelper**：节点挂到需要保存的场景子节点，`save_properties` 数组指定属性
- **Dynamic scenes**：动态实例化的子场景（如刷新的敌人）也能捕获状态
- **Global state**：跨场景传递的状态（如触发事件影响另一场景）
- **Checkpoint**：触发区域自动保存，带磁盘图标动画
- **MD5 hash**：`.json` + `.dat` 校验文件完整性

### 3. kimbunner/godot_save（Addon Save）
- **AES-256 + ZIP压缩**
- **自动备份**：保留最近N份
- **自动保存Timer**：可配置间隔
- **截图队列**：保存时自动截屏作为存档缩略图

## 核心设计

### 存档分三类

| 类型 | 文件 | 说明 |
|------|------|------|
| **全局存档** | `user://global.save` | 解锁角色、总金币、成就、PVP记录、设置 |
| **RUN存档** | `user://slot_1.save` ~ `user://slot_3.save` | 爬塔进度、队伍、层数、金币、道具 |
| **元数据** | `user://slot_1.meta` ~ `user://slot_3.meta` | 明文，槽位列表、时间戳、层数、缩略图路径 |

```gdscript
## save_manager.gd — Autoload
extends Node

signal save_completed(slot_id: int, success: bool)
signal load_completed(slot_id: int, success: bool)
signal slot_deleted(slot_id: int)
signal auto_save_triggered(slot_id: int)

const SAVE_FOLDER := "user://saves/"
const GLOBAL_FILE := "global.save"
const RUN_PREFIX := "slot_"
const META_SUFFIX := ".meta"
const SAVE_SUFFIX := ".save"
const BACKUP_SUFFIX := ".backup"
const TEMP_SUFFIX := ".tmp"

const CURRENT_SCHEMA_VERSION := 1
const ENCRYPTION_PASSWORD := "HeroBattle2025"  # 项目密码，硬编码
const MAX_SLOTS := 3
const AUTO_SAVE_INTERVAL_SECONDS := 300  # 5分钟自动保存

var _current_slot_id: int = -1      # -1 = 无进行中的RUN
var _auto_save_timer: Timer = null
var _is_saving: bool = false

func _ready() -> void:
    _ensure_folder()
    _setup_auto_save_timer()
    _migrate_old_saves()

func _ensure_folder() -> void:
    var dir := DirAccess.open("user://")
    if dir != null and not dir.dir_exists(SAVE_FOLDER):
        dir.make_dir(SAVE_FOLDER)

func _setup_auto_save_timer() -> void:
    _auto_save_timer = Timer.new()
    _auto_save_timer.wait_time = AUTO_SAVE_INTERVAL_SECONDS
    _auto_save_timer.autostart = true
    _auto_save_timer.timeout.connect(_on_auto_save)
    add_child(_auto_save_timer)

func _on_auto_save() -> void:
    if _current_slot_id >= 1 and _current_slot_id <= MAX_SLOTS:
        save_run(_current_slot_id, true)
        auto_save_triggered.emit(_current_slot_id)

## ========== 全局存档 ==========

func save_global(data: Dictionary) -> bool:
    data["schema_version"] = CURRENT_SCHEMA_VERSION
    return _write_encrypted(SAVE_FOLDER + GLOBAL_FILE, data)

func load_global() -> Dictionary:
    var data := _read_encrypted(SAVE_FOLDER + GLOBAL_FILE)
    if data.is_empty():
        return _create_default_global()
    
    # Schema版本检查
    var schema: int = data.get("schema_version", 0)
    if schema < CURRENT_SCHEMA_VERSION:
        data = _migrate_global(data, schema)
    
    return data

func _create_default_global() -> Dictionary:
    return {
        "schema_version": CURRENT_SCHEMA_VERSION,
        "unlocked_heroes": ["hero_1"],
        "total_gold": 0,
        "total_runs": 0,
        "total_wins": 0,
        "achievements": {},
        "pvp_stats": {"wins": 0, "losses": 0, "rating": 1000},
        "settings": {
            "music_volume": 0.8,
            "sfx_volume": 1.0,
            "screen_resolution": "1920x1080",
            "fullscreen": true,
        },
        "last_used_slot": 1,
    }

## ========== RUN存档 ==========

func start_new_run(slot_id: int, run_data: Dictionary) -> bool:
    if slot_id < 1 or slot_id > MAX_SLOTS:
        push_error("Invalid slot ID: %d" % slot_id)
        return false
    
    _current_slot_id = slot_id
    run_data["slot_id"] = slot_id
    run_data["save_time"] = _get_timestamp()
    run_data["play_time_seconds"] = 0
    
    var success := save_run(slot_id, false)
    if success:
        _update_meta(slot_id, run_data)
    
    return success

func save_run(slot_id: int, is_auto: bool = false) -> bool:
    if slot_id < 1 or slot_id > MAX_SLOTS:
        return false
    if _is_saving:
        return false  # 防止并发保存
    
    _is_saving = true
    
    # 1. 收集当前RUN数据
    var run_data := _collect_run_data()
    run_data["slot_id"] = slot_id
    run_data["save_time"] = _get_timestamp()
    run_data["is_auto_save"] = is_auto
    
    # 2. 增加游戏时长
    var existing := load_run(slot_id)
    if not existing.is_empty():
        var additional_time := _calculate_session_time()
        run_data["play_time_seconds"] = existing.get("play_time_seconds", 0) + additional_time
    
    # 3. 原子写入
    var file_path := SAVE_FOLDER + RUN_PREFIX + str(slot_id) + SAVE_SUFFIX
    var success := _write_encrypted(file_path, run_data)
    
    # 4. 备份旋转
    if success:
        _rotate_backup(file_path)
        _update_meta(slot_id, run_data)
        _capture_thumbnail(slot_id)
    
    _is_saving = false
    save_completed.emit(slot_id, success)
    
    return success

func load_run(slot_id: int) -> Dictionary:
    if slot_id < 1 or slot_id > MAX_SLOTS:
        return {}
    
    var file_path := SAVE_FOLDER + RUN_PREFIX + str(slot_id) + SAVE_SUFFIX
    var data := _read_encrypted(file_path)
    
    if data.is_empty():
        return {}
    
    # Schema版本检查
    var schema: int = data.get("schema_version", 0)
    if schema < CURRENT_SCHEMA_VERSION:
        data = _migrate_run(data, schema)
    
    _current_slot_id = slot_id
    return data

func delete_run(slot_id: int) -> bool:
    if slot_id < 1 or slot_id > MAX_SLOTS:
        return false
    
    var base := SAVE_FOLDER + RUN_PREFIX + str(slot_id)
    var files := [base + SAVE_SUFFIX, base + META_SUFFIX, base + BACKUP_SUFFIX, base + "_thumb.png"]
    
    for file in files:
        if FileAccess.file_exists(file):
            DirAccess.remove_absolute(file)
    
    slot_deleted.emit(slot_id)
    return true

## ========== 元数据（明文，槽位列表用） ==========

func get_all_slot_meta() -> Array[Dictionary]:
    var result: Array[Dictionary] = []
    for i in range(1, MAX_SLOTS + 1):
        result.append(get_slot_meta(i))
    return result

func get_slot_meta(slot_id: int) -> Dictionary:
    var meta_path := SAVE_FOLDER + RUN_PREFIX + str(slot_id) + META_SUFFIX
    
    if FileAccess.file_exists(meta_path):
        var file := FileAccess.open(meta_path, FileAccess.READ)
        if file != null:
            var content := file.get_as_text()
            file.close()
            var parsed := JSON.parse_string(content)
            if parsed is Dictionary:
                return parsed
    
    # 返回空占位
    return {
        "slot_id": slot_id,
        "exists": false,
        "save_time": "",
        "play_time_formatted": "",
        "hero_name": "",
        "current_floor": 0,
        "difficulty": "",
        "thumbnail_path": "",
    }

func _update_meta(slot_id: int, run_data: Dictionary) -> void:
    var meta := {
        "slot_id": slot_id,
        "exists": true,
        "save_time": run_data.get("save_time", ""),
        "play_time_formatted": _format_time(run_data.get("play_time_seconds", 0)),
        "hero_name": _get_hero_name(run_data.get("hero_id", "")),
        "current_floor": run_data.get("current_floor", 1),
        "difficulty": "老手" if run_data.get("is_veteran_mode", false) else "新手",
        "thumbnail_path": SAVE_FOLDER + RUN_PREFIX + str(slot_id) + "_thumb.png",
    }
    
    var meta_path := SAVE_FOLDER + RUN_PREFIX + str(slot_id) + META_SUFFIX
    var file := FileAccess.open(meta_path, FileAccess.WRITE)
    if file != null:
        file.store_string(JSON.stringify(meta, "\t"))
        file.close()

## ========== 底层读写（加密+原子写入） ==========

func _write_encrypted(file_path: String, data: Dictionary) -> bool:
    # 1. JSON序列化
    var json := JSON.stringify(data, "\t")
    
    # 2. 计算HMAC签名（简化版：MD5）
    var signature := json.md5_text()
    var payload := {
        "signature": signature,
        "data": data,
    }
    var final_json := JSON.stringify(payload)
    
    # 3. 临时文件写入
    var temp_path := file_path + TEMP_SUFFIX
    var file := FileAccess.open_encrypted_with_pass(temp_path, FileAccess.WRITE, ENCRYPTION_PASSWORD)
    if file == null:
        push_error("Failed to open temp file for writing: %s" % temp_path)
        return false
    
    file.store_string(final_json)
    file.close()
    
    # 4. 备份旧文件
    if FileAccess.file_exists(file_path):
        var backup_path := file_path + BACKUP_SUFFIX
        if FileAccess.file_exists(backup_path):
            DirAccess.remove_absolute(backup_path)
        DirAccess.rename_absolute(file_path, backup_path)
    
    # 5. 原子重命名
    var error := DirAccess.rename_absolute(temp_path, file_path)
    if error != OK:
        push_error("Failed to rename temp file: %d" % error)
        return false
    
    return true

func _read_encrypted(file_path: String) -> Dictionary:
    if not FileAccess.file_exists(file_path):
        return {}
    
    var file := FileAccess.open_encrypted_with_pass(file_path, FileAccess.READ, ENCRYPTION_PASSWORD)
    if file == null:
        push_error("Failed to open file for reading: %s" % file_path)
        return {}
    
    var content := file.get_as_text()
    file.close()
    
    var parsed := JSON.parse_string(content)
    if parsed == null or not (parsed is Dictionary):
        push_error("Invalid save file format: %s" % file_path)
        return {}
    
    # 校验签名
    var stored_sig := parsed.get("signature", "")
    var data := parsed.get("data", {})
    var expected_sig := JSON.stringify(data).md5_text()
    
    if stored_sig != expected_sig:
        push_warning("Save file signature mismatch: %s" % file_path)
        # 尝试从备份恢复
        return _try_backup_recovery(file_path)
    
    return data

func _try_backup_recovery(file_path: String) -> Dictionary:
    var backup_path := file_path + BACKUP_SUFFIX
    if FileAccess.file_exists(backup_path):
        push_warning("Attempting backup recovery: %s" % backup_path)
        return _read_encrypted(backup_path)
    return {}

func _rotate_backup(file_path: String) -> void:
    # 保留最近1个备份，如需更多可扩展
    var backup_path := file_path + BACKUP_SUFFIX
    if FileAccess.file_exists(backup_path):
        DirAccess.remove_absolute(backup_path)
    if FileAccess.file_exists(file_path):
        DirAccess.copy_absolute(file_path, backup_path)

func _capture_thumbnail(slot_id: int) -> void:

## ========== 辅助函数 ==========

func _get_timestamp() -> String:
    var dt := Time.get_datetime_dict_from_system()
    return "%04d-%02d-%02dT%02d:%02d:%02d" % [dt.year, dt.month, dt.day, dt.hour, dt.minute, dt.second]

func _format_time(seconds: int) -> String:
    var mins := seconds / 60
    var secs := seconds % 60
    return "%02d:%02d" % [mins, secs]

func _get_hero_name(hero_id: String) -> String:
    var config := ConfigManager.get_hero_config(hero_id)
    return config.get("hero_name", hero_id)

func _calculate_session_time() -> int:
    # 基于上一次保存时间计算本次会话时长
    # 简化实现：GameManager记录 _last_save_timestamp
    return GameManager.get_session_time_seconds()

## ========== Schema迁移 ==========

func _migrate_global(data: Dictionary, old_schema: int) -> Dictionary:
    if old_schema < 1:
        # v0 → v1：添加 settings 字段
        data["settings"] = {
            "music_volume": 0.8,
            "sfx_volume": 1.0,
            "screen_resolution": "1920x1080",
            "fullscreen": true,
        }
    
    data["schema_version"] = CURRENT_SCHEMA_VERSION
    return data

func _migrate_run(data: Dictionary, old_schema: int) -> Dictionary:
    if old_schema < 1:
        # v0 → v1：添加 play_time_seconds
        data["play_time_seconds"] = 0
    
    data["schema_version"] = CURRENT_SCHEMA_VERSION
    return data

func _migrate_old_saves() -> void:
    # 启动时检查并迁移旧格式存档
    # 如果检测到旧版本（无schema_version字段），自动迁移
    var global := load_global()
    if not global.is_empty() and global.get("schema_version", 0) < CURRENT_SCHEMA_VERSION:
        save_global(global)
    
    for i in range(1, MAX_SLOTS + 1):
        var run := load_run(i)
        if not run.is_empty() and run.get("schema_version", 0) < CURRENT_SCHEMA_VERSION:
            _write_encrypted(SAVE_FOLDER + RUN_PREFIX + str(i) + SAVE_SUFFIX, run)
```

## Step 6：解锁进度持久化

```gdscript
## 战斗胜利/成就解锁时更新全局存档

func _on_run_completed(victory: bool, hero_id: String, floor_reached: int) -> void:
    var global := SaveManager.load_global()
    
    global["total_runs"] = global.get("total_runs", 0) + 1
    global["total_gold"] = global.get("total_gold", 0) + GameManager.gold
    
    if victory:
        global["total_wins"] = global.get("total_wins", 0) + 1
        
        # 检查通关解锁
        if floor_reached >= 5 and not global["achievements"].get("clear_floor_5", false):
            global["achievements"]["clear_floor_5"] = true
            _show_unlock_notification("成就解锁：征服第5层！")
        
        # 角色通关次数
        var hero_wins_key := "hero_%s_wins" % hero_id
        global["achievements"][hero_wins_key] = global["achievements"].get(hero_wins_key, 0) + 1
    
    SaveManager.save_global(global)

func _on_hero_unlocked(hero_id: String) -> void:
    var global := SaveManager.load_global()
    var unlocked: Array = global.get("unlocked_heroes", [])
    if not unlocked.has(hero_id):
        unlocked.append(hero_id)
        global["unlocked_heroes"] = unlocked
        SaveManager.save_global(global)
        _show_unlock_notification("新角色解锁：%s！" % _get_hero_name(hero_id))
```
