# 局外商店任务卡（魔城币解锁系统）— 账号绑定版

## 重要前提：账号绑定架构
当前阶段用本地文件实现，但**数据结构必须预留账号绑定字段**，后续接入登录系统时零成本迁移。

### 当前实现（本地文件）
```json
{
  "user_id": "local_default",  // ← 预留，后续替换为真实用户ID
  "mocheng_coin": 120,
  "unlocked_heroes": [1, 2],
  "unlocked_partners": [101],
  "unlocked_skins": [],
  "last_save": 1234567890
}
```

### 未来迁移（服务器对接）
只需把 `user_id` 从 `"local_default"` 改为真实用户ID，数据结构和存储逻辑不变。

## 参考方案
基于 OpenClaw-lHq 的 GitHub 搜索分析，选用 **gloot 轻量方案**：
- 商品定义表用 JSON/protoset
- 解锁状态持久化到 `user://meta_progression.json`
- 魔城币已存在于 `RunSnapshot` 和 `game_manager.gd`

## 当前已有基础
- `shop.gd` / `shop_popup.gd` — 局外商店框架（伙伴解锁）
- `game_manager.gd` — `pending_mocheng_coin` / `mocheng_coin_total`
- `config_manager.gd` — `get_mocheng_coin_total()` / `set_mocheng_coin_total()`
- `save_manager.gd` — `save_mocheng_coin()` / `load_mocheng_coin()`

## 扩展目标
1. **账号绑定字段预留**：所有持久化数据包含 `user_id`
2. 英雄解锁（当前只有伙伴）

### Step 2：魔城币存储（账号隔离）

```gdscript
func save_mocheng_coin(amount: int, user_id: String = current_user_id) -> bool:
    var file_path: String = "user://%s_mocheng_coin.json" % user_id
    var data: Dictionary = {
        "user_id": user_id,
        "amount": amount,
        "timestamp": Time.get_unix_time_from_system()
    }
    var file := FileAccess.open(file_path, FileAccess.WRITE)
    if file == null:
        return false
    file.store_string(JSON.stringify(data))
    file.close()
    return true

func load_mocheng_coin(user_id: String = current_user_id) -> int:
    var file_path: String = "user://%s_mocheng_coin.json" % user_id
    if not FileAccess.file_exists(file_path):
        return 0
    var file := FileAccess.open(file_path, FileAccess.READ)
    var json := JSON.new()
    var result := json.parse(file.get_as_text())
    file.close()
    if result == OK:
        var data: Dictionary = json.get_data()
        # 验证 user_id 匹配
        if data.get("user_id", "") == user_id:
            return data.get("amount", 0)
    return 0
```

### Step 3：解锁状态存储（账号隔离）

```gdscript
func save_unlock_state(
    unlocked_heroes: Array[int], 
    unlocked_partners: Array[int], 
    unlocked_skins: Array[int],
    user_id: String = current_user_id
) -> bool:
    var file_path: String = "user://%s_meta_progression.json" % user_id
    var data: Dictionary = {
        "user_id": user_id,
        "unlocked_heroes": unlocked_heroes,
        "unlocked_partners": unlocked_partners,
        "unlocked_skins": unlocked_skins,
        "last_save": Time.get_unix_time_from_system()
    }
    var file := FileAccess.open(file_path, FileAccess.WRITE)
    if file == null:
        return false
    file.store_string(JSON.stringify(data))
    file.close()
    print("[SaveManager] 解锁状态已保存: %s" % file_path)
    return true

func load_unlock_state(user_id: String = current_user_id) -> Dictionary:
    var file_path: String = "user://%s_meta_progression.json" % user_id
    if not FileAccess.file_exists(file_path):
        # 默认只解锁英雄1（战士）
        return {
            "user_id": user_id,
            "unlocked_heroes": [1],
            "unlocked_partners": [],
            "unlocked_skins": [],
            "last_save": 0
        }
    var file := FileAccess.open(file_path, FileAccess.READ)
    var json := JSON.new()
    var result := json.parse(file.get_as_text())
    file.close()
    if result == OK:
        var data: Dictionary = json.get_data()
        # 验证 user_id 匹配，不匹配则返回默认
        if data.get("user_id", "") == user_id:
            return data
    return {
        "user_id": user_id,
        "unlocked_heroes": [1],
        "unlocked_partners": [],
        "unlocked_skins": [],
        "last_save": 0
    }
```

### Step 5：局外商店面板

**文件：`scenes/shop/shop_popup.gd`**

```gdscript
extends Panel

@onready var tab_container: TabContainer = $TabContainer
@onready var coin_label: Label = $CoinLabel
@onready var user_label: Label = $UserLabel  # 显示当前用户ID

var _shop_items: Dictionary = {}
var _unlocked_heroes: Array[int] = []
var _unlocked_partners: Array[int] = []
var _unlocked_skins: Array[int] = []
var _current_coin: int = 0
var _current_user_id: String = "local_default"

signal item_purchased(item_id: int, item_type: String)
signal closed

func _ready() -> void:
    _current_user_id = SaveManager.get_user_id()
    _load_shop_items()
    _load_unlock_state()
    _update_coin_display()
    _render_all_tabs()

func _load_shop_items() -> void:
    var file := FileAccess.open("res://resources/configs/shop_items.json", FileAccess.READ)
    var json := JSON.new()
    json.parse(file.get_as_text())
    _shop_items = json.get_data() as Dictionary

func _load_unlock_state() -> void:
    var state := SaveManager.load_unlock_state(_current_user_id)
    _unlocked_heroes = state.get("unlocked_heroes", [1])
    _unlocked_partners = state.get("unlocked_partners", [])
    _unlocked_skins = state.get("unlocked_skins", [])
    _current_coin = SaveManager.load_mocheng_coin(_current_user_id)

func _update_coin_display() -> void:
    coin_label.text = "💰 %d" % _current_coin
    user_label.text = "用户: %s" % _current_user_id  # 调试用，后续可隐藏

func _render_all_tabs() -> void:
    _render_tab("heroes", _shop_items.get("heroes", []), _unlocked_heroes)
    _render_tab("partners", _shop_items.get("partners", []), _unlocked_partners)
    _render_tab("skins", _shop_items.get("skins", []), _unlocked_skins)

func _render_tab(tab_name: String, items: Array, unlocked: Array[int]) -> void:
    var grid: GridContainer = get_node("TabContainer/%s/ItemGrid" % tab_name.capitalize())
    for child in grid.get_children():
        child.queue_free()
    for item in items:
        var button := _create_item_button(item, unlocked)
        grid.add_child(button)

func _create_item_button(item: Dictionary, unlocked: Array[int]) -> Button:
    var btn := Button.new()
    var id: int = item.get("id", 0)
    var cost: int = item.get("cost", 0)
    var is_owned: bool = id in unlocked
    var can_afford: bool = _current_coin >= cost
    
    btn.text = "%s\n💰 %d" % [item.get("name", "???"), cost]
    btn.custom_minimum_size = Vector2(120, 160)
    
    if is_owned:
        btn.text += "\n✅ 已拥有"
        btn.modulate = Color(0.5, 0.5, 0.5)
        btn.disabled = true
    elif not can_afford:
        btn.modulate = Color(0.7, 0.3, 0.3)
        btn.disabled = true
    else:
        btn.pressed.connect(_on_purchase_requested.bind(item))
    return btn

func _on_purchase_requested(item: Dictionary) -> void:
    var id: int = item.get("id", 0)
    var cost: int = item.get("cost", 0)
    var type: String = _get_item_type(id)
    
    var confirm := AcceptDialog.new()
    confirm.title = "确认购买"
    confirm.dialog_text = "花费 %d 魔城币购买 %s？" % [cost, item.get("name", "???")]
    confirm.confirmed.connect(_confirm_purchase.bind(item))
    add_child(confirm)
    confirm.popup_centered()

func _confirm_purchase(item: Dictionary) -> void:
    var id: int = item.get("id", 0)
    var cost: int = item.get("cost", 0)
    var type: String = _get_item_type(id)
    
    if _current_coin < cost:
        return
    
    _current_coin -= cost
    SaveManager.save_mocheng_coin(_current_coin, _current_user_id)
    
    match type:
        "hero":
            _unlocked_heroes.append(id)
        "partner":
            _unlocked_partners.append(id)
        "skin":
            _unlocked_skins.append(id)
    
    SaveManager.save_unlock_state(_unlocked_heroes, _unlocked_partners, _unlocked_skins, _current_user_id)
    item_purchased.emit(id, type)
    _update_coin_display()
    _render_all_tabs()

func _get_item_type(id: int) -> String:
    if id < 100:
        return "hero"
    elif id < 1000:
        return "partner"
    else:
        return "skin"

func _on_close_button_pressed() -> void:
    closed.emit()
    visible = false
```
