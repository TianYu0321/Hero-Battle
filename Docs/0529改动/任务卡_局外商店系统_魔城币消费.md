# 任务卡：局外商店系统（Meta Shop — 魔城币消费）

## 参考来源

1. **DesirePathGames/Slay-The-Robot** — Godot 4 Roguelike Deckbuilder 框架，内置 Shop / Rest / Chest / Consumables 系统，Action-driven 购买流程
2. **thedivergentai/gd-agentic-skills** — `godot-economy-system` 技能：多货币、商店系统、动态定价、战利品表
3. **MarvinTMavee/Godot-ShopSystem** — Godot 4.1 简单商店系统，含状态标签更新、购买确认流程

## 核心规则（上下文衔接）

| 规则 | 说明 |
|------|------|
| **货币** | 魔城币（来自 PVP 胜利奖励，每日限领5次） |
| **商店类型** | 局外商店（Meta Shop），永久升级，不随单次 RUN 重置 |
| **购买限制** | 部分商品有购买次数上限；已拥有的商品不可重复购买 |
| **生效方式** | 局外升级在「开始新 RUN」时生效，影响初始状态 |
| **入口** | 主菜单新增「商店」按钮；PVP 大厅也可跳转 |

## Step 1：MetaShopManager（Autoload Singleton）

```gdscript
## autoload/meta_shop_manager.gd
extends Node

signal item_purchased(item_id: String, new_level: int)
signal insufficient_funds(required: int, current: int)
signal shop_data_changed

const SAVE_KEY := "meta_upgrades"

## ========== 初始化/加载 ==========

func _ready() -> void:
    _ensure_save_structure()

func _ensure_save_structure() -> void:
    var global: Dictionary = SaveManager.load_global()
    if not global.has(SAVE_KEY):
        global[SAVE_KEY] = {
            "starting_gold_bonus": {"level": 0, "max_level": 5, "cost_per_level": 50},
            "starting_hp_bonus": {"level": 0, "max_level": 3, "cost_per_level": 80},
            "permanent_atk_bonus": {"level": 0, "max_level": 5, "cost_per_level": 120},
            "unlocked_partners": ["partner_1", "partner_2"],
            "pvp_daily_reward_bonus": {"level": 0, "max_level": 3, "cost_per_level": 300},
            "shop_discount": {"purchased": false, "cost": 200},
        }
        SaveManager.save_global(global)

## ========== 查询接口（UI 用）==========

func get_upgrade_data(item_id: String) -> Dictionary:
    var global: Dictionary = SaveManager.load_global()
    var meta: Dictionary = global.get(SAVE_KEY, {})
    
    ## 特殊处理：解锁类
    if item_id.begins_with("unlock_partner_"):
        var target: String = item_id.replace("unlock_partner_", "partner_")
        var unlocked: Array = meta.get("unlocked_partners", [])
        return {
            "level": 1 if target in unlocked else 0,
            "max_level": 1,
            "purchased": target in unlocked,
        }
    
    ## 特殊处理：一次性购买
    if item_id == "shop_discount":
        var data: Dictionary = meta.get("shop_discount", {})
        return {
            "level": 1 if data.get("purchased", false) else 0,
            "max_level": 1,
            "purchased": data.get("purchased", false),
        }
    
    ## 等级类
    return meta.get(item_id, {"level": 0, "max_level": 1})

func get_current_cost(item_id: String) -> int:
    var item: Dictionary = MetaShopData.get_item(item_id)
    var upgrade_data: Dictionary = get_upgrade_data(item_id)
    var current_level: int = upgrade_data.get("level", 0)
    return MetaShopData.get_current_cost(item_id, current_level)

func can_purchase(item_id: String) -> bool:
    var upgrade_data: Dictionary = get_upgrade_data(item_id)
    var current_level: int = upgrade_data.get("level", 0)
    var max_level: int = upgrade_data.get("max_level", 1)
    
    if current_level >= max_level:
        return false  # 已达上限
    
    var cost: int = get_current_cost(item_id)
    var magic_coins: int = PVPManager.get_stats().get("magic_coins", 0)
    return magic_coins >= cost

## ========== 购买 ==========

func purchase(item_id: String) -> bool:
    if not can_purchase(item_id):
        var cost: int = get_current_cost(item_id)
        var magic_coins: int = PVPManager.get_stats().get("magic_coins", 0)
        insufficient_funds.emit(cost, magic_coins)
        return false
    
    var cost: int = get_current_cost(item_id)
    var global: Dictionary = SaveManager.load_global()
    var meta: Dictionary = global.get(SAVE_KEY, {})
    
    ## 扣除魔城币（通过 PVPManager）
    if not PVPManager.spend_magic_coins(cost):
        insufficient_funds.emit(cost, PVPManager.get_stats().get("magic_coins", 0))
        return false
    
    ## 处理不同类型
    if item_id.begins_with("unlock_partner_"):
        var target: String = item_id.replace("unlock_partner_", "partner_")
        var unlocked: Array = meta.get("unlocked_partners", [])
        if target not in unlocked:
            unlocked.append(target)
            meta["unlocked_partners"] = unlocked
    elif item_id == "shop_discount":
        meta["shop_discount"]["purchased"] = true
    else:
        ## 等级类
        var data: Dictionary = meta.get(item_id, {"level": 0})
        data["level"] = data.get("level", 0) + 1
        meta[item_id] = data
    
    global[SAVE_KEY] = meta
    SaveManager.save_global(global)
    
    item_purchased.emit(item_id, get_upgrade_data(item_id).get("level", 0))
    shop_data_changed.emit()
    return true

## ========== 生效接口（GameManager 在 RUN 开始时调用）==========

## 获取初始金币加成
func get_starting_gold_bonus() -> int:
    var data: Dictionary = get_upgrade_data("starting_gold_bonus")
    return data.get("level", 0) * 10

## 获取初始生命加成
func get_starting_hp_bonus() -> int:
    var data: Dictionary = get_upgrade_data("starting_hp_bonus")
    return data.get("level", 0) * 5

## 获取永久攻击力加成
func get_permanent_atk_bonus() -> int:
    var data: Dictionary = get_upgrade_data("permanent_atk_bonus")
    return data.get("level", 0)

## 获取已解锁伙伴列表
func get_unlocked_partners() -> Array:
    var global: Dictionary = SaveManager.load_global()
    var meta: Dictionary = global.get(SAVE_KEY, {})
    return meta.get("unlocked_partners", ["partner_1", "partner_2"])

## 获取 PVP 每日奖励额外次数
func get_pvp_daily_reward_bonus() -> int:
    var data: Dictionary = get_upgrade_data("pvp_daily_reward_bonus")
    return data.get("level", 0)

## 获取局内商店折扣
func get_shop_discount() -> float:
    var data: Dictionary = get_upgrade_data("shop_discount")
    if data.get("purchased", false):
        return 0.9  # 9折
    return 1.0  # 无折扣

## ========== PVPManager 联动 ==========

## 获取 PVP 每日最大奖励次数（基础5次 + 升级加成）
func get_total_pvp_daily_max() -> int:
    return 5 + get_pvp_daily_reward_bonus()
```

```gdscript
func _setup_category_tabs() -> void:
    for cat_key in MetaShopData.CATEGORIES.keys():
        var btn := Button.new()
        btn.text = MetaShopData.CATEGORIES[cat_key]
        btn.toggle_mode = true
        btn.button_pressed = (cat_key == _current_category)
        _apply_tab_style(btn, cat_key == _current_category)

func _on_category_selected(cat: String) -> void:
    _current_category = cat
    for i in range(category_tabs.get_child_count()):
        var btn: Button = category_tabs.get_child(i)
        var is_active: bool = MetaShopData.CATEGORIES.keys()[i] == cat
        btn.button_pressed = is_active
        _apply_tab_style(btn, is_active)
    
    _update_item_grid()
    _select_first_item()
    AudioManager.play_ui("click")

func _update_item_grid() -> void:
    ## 清除旧卡片
    for child in item_grid.get_children():
        child.queue_free()
    _shop_item_cards.clear()
    
    for item in MetaShopData.SHOP_ITEMS:
        if item["category"] != _current_category:

        var card := _create_item_card(item)
        item_grid.add_child(card)
        _shop_item_cards[item["id"]] = card
    
    ## 入场动画
    var idx := 0
    for child in item_grid.get_children():
        child.modulate.a = 0.0
        child.position.y += 20
        var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
        tween.tween_property(child, "modulate:a", 1.0, 0.3).set_delay(idx * 0.05)
        tween.parallel().tween_property(child, "position:y", child.position.y - 20, 0.3).set_delay(idx * 0.05)
        idx += 1

func _select_item(item_id: String) -> void:
    _selected_item_id = item_id
    var item: Dictionary = MetaShopData.get_item(item_id)
    var upgrade_data: Dictionary = MetaShopManager.get_upgrade_data(item_id)
    var current_level: int = upgrade_data.get("level", 0)
    var max_level: int = upgrade_data.get("max_level", 1)
    var is_maxed: bool = current_level >= max_level
    
    detail_name.text = item["name"]
    detail_name.add_theme_font_size_override("font_size", 20)
    detail_desc.text = item["description"]
    detail_desc.add_theme_font_size_override("font_size", 13)
    
    ## 当前效果
    if is_maxed:
        effect_label.text = "当前效果：" + item["effect_description"] % current_level
        next_effect_label.text = "已达最高等级"
        next_effect_label.add_theme_color_override("font_color", Color(0.6, 0.6, 0.62, 1))
    else:
        effect_label.text = "当前效果：" + item["effect_description"] % current_level
        next_effect_label.text = "升级后：" + item["effect_description"] % (current_level + 1)
        next_effect_label.add_theme_color_override("font_color", Color(0.25, 0.55, 0.9, 1))
    
    ## 价格
    if is_maxed:
        cost_value.text = "—"
        purchase_btn.disabled = true
        purchase_btn.text = "已拥有"
        _apply_disabled_button_style(purchase_btn)
    else:
        var cost: int = MetaShopManager.get_current_cost(item_id)
        cost_value.text = "%d" % cost
        
        var can_afford: bool = MetaShopManager.can_purchase(item_id)
        purchase_btn.disabled = not can_afford
        purchase_btn.text = "购买" if can_afford else "魔城币不足"
        _apply_primary_button_style(purchase_btn, can_afford)
    
    AudioManager.play_ui("click")

func _on_item_purchased(item_id: String, new_level: int) -> void:
    ## 更新选中项详情

        _select_item(item_id)

    if _shop_item_cards.has(item_id):
        var card: PanelContainer = _shop_item_cards[item_id]
        ## 重新生成该卡片（最简单方式）

        item_grid.move_child(new_card, card.get_index())

        _shop_item_cards[item_id] = new_card

    _show_purchase_success()
    _update_magic_coins()
    _update_bottom_bonus()
    AudioManager.play_ui("success")
```

## Step 5：GameManager RUN 启动时应用局外加成

```gdscript
## game_manager.gd — start_new_run() 中

func start_new_run(hero_id: String, partner_ids: Array) -> void:
    ## ... 原有初始化 ...
    
    ## === 应用局外商店加成 ===
    
    ## 初始金币加成
    var gold_bonus: int = MetaShopManager.get_starting_gold_bonus()
    run_data["gold"] = 100 + gold_bonus  # 基础100 + 加成
    
    ## 初始生命加成
    var hp_bonus: int = MetaShopManager.get_starting_hp_bonus()
    run_data["hero_max_hp"] = run_data.get("hero_max_hp", 80) + hp_bonus
    run_data["hero_hp"] = run_data["hero_max_hp"]
    
    ## 永久攻击力加成（存入 run_data，战斗时读取）
    var atk_bonus: int = MetaShopManager.get_permanent_atk_bonus()
    run_data["permanent_atk_bonus"] = atk_bonus
    
    ## 局内商店折扣
    run_data["shop_discount"] = MetaShopManager.get_shop_discount()
    
    ## 过滤未解锁的伙伴
    var unlocked_partners: Array = MetaShopManager.get_unlocked_partners()
    var valid_partners: Array = []
    for pid in partner_ids:
        if pid in unlocked_partners:
            valid_partners.append(pid)
    ## 确保至少有默认伙伴
    if valid_partners.is_empty():
        valid_partners = ["partner_1"]
    run_data["partner_ids"] = valid_partners
    
    SaveManager.save_run(current_slot, run_data)
```
