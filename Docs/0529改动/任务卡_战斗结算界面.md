# 任务卡：战斗结算界面（胜利 / 失败）

## 参考来源

1. **ptrlrd/spire-codex** — Shared Run 页面：in-game-style victory/defeat summary，含层数节点、relic strip、统计汇总
2. **hi-godot/save-system-godot-claude** — 胜利面板（victory panel）+ 生涯统计（career stats footer）
3. **CraftPix Free Basic Pixel Art UI for RPG** — victory/defeat screens 基础布局参考（Label + 统计行 + 按钮组）

---

## 核心规则

| 场景 | 说明 |
|------|------|
| **胜利** | 敌方全部阵亡，显示获得奖励（金币/掉落物），提供「继续爬塔」/「返回主菜单」 |
| **失败** | 我方主角 HP=0，显示本次 Run 统计（到达层数/击败敌人数/累计金币等），提供「重新开始」/「返回主菜单」 |
| **PVP 胜利/失败** | 已有 PVPManager 结算面板，本次不覆盖 |

## Step 0：结算数据结构

```gdscript
## 战斗结束时由 battle_scene.gd 生成，传递给结算面板
class_name BattleResult
extends RefCounted

enum Outcome { VICTORY, DEFEAT }

var outcome: Outcome
var run_snapshot: Dictionary = {}     # 当前 RUN 数据快照
var battle_rewards: Dictionary = {}   # 战斗奖励（仅 VICTORY）
var battle_stats: Dictionary = {}    # 战斗统计（双方通用）

## === VICTORY 时填充 ===
## battle_rewards:
##   "gold_earned": int          # 本战斗获得金币
##   "drops": Array[Dictionary]  # 掉落物列表 {name, icon, count}
##   "floor_cleared": bool      # 是否清理本层（击败层Boss）

## === DEFEAT 时填充 ===
## battle_stats:
##   "total_floors": int         # 本次Run到达的最高层数
##   "enemies_defeated": int    # 累计击败敌人数
##   "total_gold_collected": int # 累计获得金币
##   "total_battles": int       # 总战斗次数
##   "play_time": String         # 格式化时长 "12:34"
```

## Step 1：结算面板 UI 场景

### 节点结构（统一面板，根据 outcome 切换内容）


```gdscript
## scenes/battle/battle_result_panel.gd
extends CanvasLayer

signal primary_action_pressed
signal secondary_action_pressed

@onready var main_panel: PanelContainer = $MainPanel
@onready var title_label: Label = $MainPanel/VBoxContainer/OutcomeTitle
@onready var content_container: Control = $MainPanel/VBoxContainer/ContentContainer
@onready var primary_btn: Button = $MainPanel/VBoxContainer/ButtonRow/PrimaryBtn
@onready var secondary_btn: Button = $MainPanel/VBoxContainer/ButtonRow/SecondaryBtn
@onready var overlay: ColorRect = $Overlay
@onready var particles: CPUParticles2D = $ParticleEffect

var _result: BattleResult

func _ready() -> void:
    _setup_styles()
    primary_btn.pressed.connect(_on_primary)
    secondary_btn.pressed.connect(_on_secondary)
    
    ## 默认隐藏
    visible = false
    overlay.modulate.a = 0.0
    main_panel.scale = Vector2(0.85, 0.85)
    main_panel.modulate.a = 0.0

func _setup_styles() -> void:
    ## 面板样式
    var panel_style := StyleBoxFlat.new()
    panel_style.bg_color = Color(0.98, 0.98, 0.96, 1.0)
    panel_style.border_color = Color(0.8, 0.8, 0.82, 1.0)
    panel_style.border_width_left = 1
    panel_style.border_width_top = 1
    panel_style.border_width_right = 1
    panel_style.border_width_bottom = 4   # 贴纸厚度
    panel_style.corner_radius_top_left = 16
    panel_style.corner_radius_top_right = 16
    panel_style.corner_radius_bottom_left = 16
    panel_style.corner_radius_bottom_right = 16
    panel_style.shadow_color = Color(0, 0, 0, 0.12)
    panel_style.shadow_size = 12
    panel_style.shadow_offset = Vector2(0, 4)
    main_panel.add_theme_stylebox_override("panel", panel_style)
    
    ## 按钮样式
    _apply_button_style(primary_btn, true)
    _apply_button_style(secondary_btn, false)

func show_result(result: BattleResult) -> void:
    _result = result
    visible = true
    
    ## 清空旧内容
    for child in content_container.get_children():
        child.queue_free()
    
    match result.outcome:
        BattleResult.Outcome.VICTORY:
            _setup_victory(result)
        BattleResult.Outcome.DEFEAT:
            _setup_defeat(result)
    
    ## 入场动画
    overlay.modulate.a = 0.0
    main_panel.scale = Vector2(0.85, 0.85)
    main_panel.modulate.a = 0.0
    
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    tween.tween_property(overlay, "modulate:a", 1.0, 0.25)
    tween.parallel().tween_property(main_panel, "modulate:a", 1.0, 0.35)
    tween.parallel().tween_property(main_panel, "scale", Vector2.ONE, 0.35).set_trans(Tween.TRANS_BACK)

## ========== 胜利布局 ==========

func _setup_victory(result: BattleResult) -> void:
    title_label.text = "胜利！"
    title_label.add_theme_font_size_override("font_size", 32)
    title_label.add_theme_color_override("font_color", Color(0.25, 0.6, 0.35, 1))
    
    particles.emitting = true
    
    primary_btn.text = "继续冒险"
    secondary_btn.text = "返回主菜单"
    
    var vbox := VBoxContainer.new()
    vbox.add_theme_constant_override("separation", 10)
    content_container.add_child(vbox)
    
    ## 金币行
    if result.battle_rewards.get("gold_earned", 0) > 0:
        var gold_row := _create_stat_row("获得金币", "+%d" % result.battle_rewards["gold_earned"], Color(0.85, 0.65, 0.15, 1))
        vbox.add_child(gold_row)
    
    ## 掉落物网格
    var drops: Array = result.battle_rewards.get("drops", [])
    if not drops.is_empty():
        var drops_label := Label.new()
        drops_label.text = "获得物品"
        drops_label.add_theme_font_size_override("font_size", 14)
        drops_label.add_theme_color_override("font_color", Color(0.45, 0.45, 0.48, 1))
        vbox.add_child(drops_label)
        
        var drops_grid := HBoxContainer.new()
        drops_grid.add_theme_constant_override("separation", 12)
        vbox.add_child(drops_grid)
        
        for drop in drops:
            var drop_card := _create_drop_card(drop)
            drops_grid.add_child(drop_card)
    
    ## 层数清理标记
    if result.battle_rewards.get("floor_cleared", false):
        var clear_label := Label.new()
        clear_label.text = "本层已清理！前往下一层"
        clear_label.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
        clear_label.add_theme_font_size_override("font_size", 16)
        clear_label.add_theme_color_override("font_color", Color(0.25, 0.55, 0.9, 1))
        vbox.add_child(clear_label)

func _setup_defeat(result: BattleResult) -> void:

## ========== 通用辅助 ==========

func _create_stat_row(label_text: String, value_text: String, value_color: Color) -> HBoxContainer:
    var row := HBoxContainer.new()
    row.size_flags_horizontal = Control.SIZE_EXPAND_FILL
    
    var label := Label.new()
    label.text = label_text
    label.size_flags_horizontal = Control.SIZE_EXPAND_FILL
    label.add_theme_font_size_override("font_size", 14)
    label.add_theme_color_override("font_color", Color(0.45, 0.45, 0.48, 1))
    row.add_child(label)
    
    var value := Label.new()
    value.text = value_text
    value.horizontal_alignment = HORIZONTAL_ALIGNMENT_RIGHT
    value.add_theme_font_size_override("font_size", 14)
    value.add_theme_color_override("font_color", value_color)
    row.add_child(value)
    
    return row

func _create_drop_card(drop: Dictionary) -> PanelContainer:
    var card := PanelContainer.new()
    card.custom_minimum_size = Vector2(64, 80)
    
    var style := StyleBoxFlat.new()
    style.bg_color = Color(1.0, 1.0, 1.0, 1.0)
    style.border_color = Color(0.85, 0.85, 0.88, 1.0)
    style.border_width_bottom = 2
    style.corner_radius_top_left = 8
    style.corner_radius_top_right = 8
    style.corner_radius_bottom_left = 8
    style.corner_radius_bottom_right = 8
    card.add_theme_stylebox_override("panel", style)
    
    var vbox := VBoxContainer.new()
    vbox.alignment = BoxContainer.ALIGNMENT_CENTER
    card.add_child(vbox)
    
    var icon := TextureRect.new()
    icon.custom_minimum_size = Vector2(40, 40)
    icon.expand_mode = TextureRect.EXPAND_FIT_HEIGHT_PROPORTIONAL
    vbox.add_child(icon)
    
    var name_label := Label.new()
    name_label.text = drop.get("name", "???")
    name_label.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
    name_label.add_theme_font_size_override("font_size", 11)
    vbox.add_child(name_label)
    
    if drop.get("count", 1) > 1:
        var count_label := Label.new()
        count_label.text = "x%d" % drop["count"]
        count_label.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
        count_label.add_theme_font_size_override("font_size", 10)
        count_label.add_theme_color_override("font_color", Color(0.85, 0.65, 0.15, 1))
        vbox.add_child(count_label)
    
    return card

func _apply_button_style(btn: Button, primary: bool) -> void:
    var style := StyleBoxFlat.new()
    if primary:
        style.bg_color = Color(0.25, 0.55, 0.9, 1.0)
    else:
        style.bg_color = Color(0.92, 0.92, 0.94, 1.0)
    style.corner_radius_top_left = 10
    style.corner_radius_top_right = 10
    style.corner_radius_bottom_left = 10
    style.corner_radius_bottom_right = 10
    btn.add_theme_stylebox_override("normal", style)
    btn.add_theme_color_override("font_color", Color.WHITE if primary else Color(0.35, 0.35, 0.38, 1))
    btn.custom_minimum_size = Vector2(140, 44)

func _on_primary() -> void:
    _hide_panel()
    primary_action_pressed.emit()

func _on_secondary() -> void:
    _hide_panel()
    secondary_action_pressed.emit()

func _hide_panel() -> void:
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_IN)
    tween.tween_property(main_panel, "scale", Vector2(0.9, 0.9), 0.2)
    tween.parallel().tween_property(main_panel, "modulate:a", 0.0, 0.2)
    tween.parallel().tween_property(overlay, "modulate:a", 0.0, 0.25)
    tween.finished.connect(func():
        visible = false
        particles.emitting = false
    )
```

## Step 2：battle_scene.gd 集成

```gdscript
## scenes/battle/battle_scene.gd — 战斗结束时调用

@onready var result_panel: BattleResultPanel = $CanvasLayer/BattleResultPanel

func _on_battle_ended(result: Dictionary) -> void:
    if GameManager.is_pvp_mode:
        ## PVP 结算由 PVPManager 处理（已有）
        _handle_pvp_result(result)
        return
    
    var battle_result := BattleResult.new()
    
    if result.get("winner") == "player":
        ## === 胜利 ===
        battle_result.outcome = BattleResult.Outcome.VICTORY
        battle_result.battle_rewards = {
            "gold_earned": result.get("gold_reward", 10),
            "drops": result.get("drops", []),
            "floor_cleared": result.get("is_boss_fight", false),
        }
        
        ## 写入 RUN 存档
        var run_data: Dictionary = SaveManager.load_run(GameManager.current_slot)
        run_data["gold"] = run_data.get("gold", 0) + battle_result.battle_rewards["gold_earned"]
        run_data["enemies_defeated"] = run_data.get("enemies_defeated", 0) + 1
        run_data["total_battles"] = run_data.get("total_battles", 0) + 1
        
        SaveManager.save_run(GameManager.current_slot, run_data)
        
    elif result.get("winner") == "enemy":
        ## === 失败 ===
        battle_result.outcome = BattleResult.Outcome.DEFEAT
        
        var run_data: Dictionary = SaveManager.load_run(GameManager.current_slot)
        battle_result.battle_stats = {
            "total_floors": run_data.get("current_floor", 1),
            "enemies_defeated": run_data.get("enemies_defeated", 0),
            "total_gold_collected": run_data.get("total_gold_collected", run_data.get("gold", 0)),
            "total_battles": run_data.get("total_battles", 0) + 1,
            "play_time": _format_play_time(run_data.get("play_time_seconds", 0)),
        }
        
        ## 标记 RUN 结束（死亡）
        run_data["run_active"] = false
        run_data["run_ended"] = true
        run_data["run_result"] = "defeat"
        SaveManager.save_run(GameManager.current_slot, run_data)
    
    ## 显示结算面板
    result_panel.show_result(battle_result)
    result_panel.primary_action_pressed.connect(_on_result_primary.bind(battle_result.outcome))
    result_panel.secondary_action_pressed.connect(_on_result_secondary)

func _on_result_primary(outcome: BattleResult.Outcome) -> void:
    match outcome:
        BattleResult.Outcome.VICTORY:
            ## 返回爬塔场景，继续
            TransitionManager.switch_scene("res://scenes/run_main/run_main.tscn", "battle_to_run")
        BattleResult.Outcome.DEFEAT:
            ## 重新开始新 RUN
            GameManager.start_new_run(GameManager.selected_hero_id, GameManager.partner_ids)

func _on_result_secondary() -> void:
    ## 返回主菜单
    TransitionManager.switch_scene("res://scenes/menu/menu.tscn", "fade")

func _format_play_time(seconds: int) -> String:
    var mins: int = seconds / 60
    var secs: int = seconds % 60
    return "%02d:%02d" % [mins, secs]
```
