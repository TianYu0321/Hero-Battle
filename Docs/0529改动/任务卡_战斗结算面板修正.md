# 任务卡：战斗结算面板修正（失败态 + 双按钮）

## 问题

当前 `run_main.gd` 战斗失败时**不弹出结算面板**，直接走 `_run_controller.confirm_battle_result()`。
`battle_result_panel.tscn` 中 `SecondaryBtn` 默认 `visible=false`。

## 修改目标

失败时同样弹出结算面板，显示两个按钮：
- **PrimaryBtn** = "重新开始" → 启动新 RUN
- **SecondaryBtn** = "返回主菜单" → 返回主菜单

添加 `_show_defeat_panel` 方法：

```gdscript
func _show_defeat_panel(battle_result: Dictionary) -> void:
    var summary: Dictionary = _run_controller.get_run_summary() if _run_controller != null else {}
    
    var run_data: Dictionary = SaveManager.load_latest_run() if SaveManager.has_active_run() else {}
    var play_time: int = run_data.get("play_time_seconds", 0)
    
    var panel_data := {
        "outcome": "defeat",
        "battle_stats": {
            "total_floors": summary.get("max_floor", run_data.get("current_floor", 1)),
            "enemies_defeated": summary.get("enemies_defeated", 0),
            "total_gold_collected": summary.get("total_gold", run_data.get("gold", 0)),
            "total_battles": summary.get("total_battles", 0),
            "play_time": _format_play_time(play_time),
        }
    }
    
    _battle_result_panel.show_result(panel_data)
    
    ## 连接信号：重新开始 或 返回主菜单
    _battle_result_panel.primary_action_pressed.connect(
        _on_defeat_restart, CONNECT_ONE_SHOT
    )
    _battle_result_panel.secondary_action_pressed.connect(
        _on_defeat_return_to_menu, CONNECT_ONE_SHOT
    )
    print("[RunMain] 显示失败结算面板")

func _on_defeat_restart() -> void:
    print("[RunMain] 失败面板：重新开始")
    ## 清理旧 RUN 数据
    if _run_controller != null:
        _run_controller.cleanup_run()
    ## 返回主菜单，由玩家手动开始新局（或自动跳转选人界面）
    GameManager.change_scene("HERO_SELECT", "fade")

func _on_defeat_return_to_menu() -> void:
    print("[RunMain] 失败面板：返回主菜单")
    if _run_controller != null:
        _run_controller.cleanup_run()
    GameManager.change_scene("MENU", "fade")

func _format_play_time(seconds: int) -> String:
    var mins: int = seconds / 60
    var secs: int = seconds % 60
    return "%02d:%02d" % [mins, secs]
```
