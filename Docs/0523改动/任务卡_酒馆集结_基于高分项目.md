# 任务卡：酒馆集结（选取伙伴编队）— 基于真实高分项目参考

## 参考来源（已分析的真实项目）

### 1. chun92/card-framework（Godot专业卡牌框架）
- **Tween-Based Movement**：平滑、可中断动画，替代 `_process` 逐帧移动
- **Hover Effects**：scale + position 双重hover效果，含memory-safe Tween清理
- **State Machine**：拖拽状态管理，防止hover/click冲突
- **Z-Index Management**：auto-move后保持正确层级
- **Centralized Settings**：`CardFrameworkSettings` 集中配置所有常量

### 2. ycarowr/UiCard（Unity卡牌UI — Hearthstone/Slay the Spire风格）
- **Card hovered size**：悬停时卡片缩放（默认放大1.1x~1.2x）
- **Card hovered height**：悬停时Y轴上浮（上移20~40px）
- **Card hovered speed**：悬停动画速度（0.1~0.2s）
- **Disabled card Transparency**：禁用态透明度降至0.4~0.6
- **Motion speed**：scale/movement/rotation统一动画速度

### 3. IYanel-DEV/MainMenu-MP（已下载分析）
- **StyleBoxFlat shadow系统**：`shadow_size=6` + `shadow_offset=Vector2(0,2)` 制造立体感
- **White主题**：纯白底 + 浅灰边框 + 蓝色hover边框

### 4. Maaack/Godot-Menus-Template（已下载分析）
- **入场动画**：顺序淡入+上滑，Tween参数精确控制
- **子菜单封装**：`_open_sub_menu()` / `_close_sub_menu()` 模式

```gdscript
## 复用创建StyleBoxFlat，避免每个地方重复写20行
func _create_stylebox(
    bg_color: Color,
    border_color: Color,
    border_width: int,
    bottom_border: int,
    radius: int,
    shadow_size: int,
    shadow_offset: Vector2,
    shadow_color: Color
) -> StyleBoxFlat:
    var s := StyleBoxFlat.new()
    s.bg_color = bg_color
    s.border_color = border_color
    s.border_width_left = border_width
    s.border_width_top = border_width
    s.border_width_right = border_width
    s.border_width_bottom = bottom_border
    s.corner_radius_top_left = radius
    s.corner_radius_top_right = radius
    s.corner_radius_bottom_left = radius
    s.corner_radius_bottom_right = radius
    s.shadow_color = shadow_color
    s.shadow_size = shadow_size
    s.shadow_offset = shadow_offset
    return s

func _create_card_style(state: String) -> StyleBoxFlat:
    match state:
        "normal":
            return _create_stylebox(
                PartyAssembleSettings.COLOR_BG_PANEL,
                PartyAssembleSettings.COLOR_BORDER, 2, 3,
                PartyAssembleSettings.RADIUS_CARD,
                PartyAssembleSettings.SHADOW_CARD_NORMAL.size,

                PartyAssembleSettings.SHADOW_CARD_NORMAL.color
            )
        "hover":
            return _create_stylebox(
                PartyAssembleSettings.COLOR_BG_PANEL,
                PartyAssembleSettings.COLOR_BORDER_HOVER, 2, 3,
                PartyAssembleSettings.RADIUS_CARD,
                PartyAssembleSettings.SHADOW_CARD_HOVER.size,
                PartyAssembleSettings.SHADOW_CARD_HOVER.offset,
                PartyAssembleSettings.SHADOW_CARD_HOVER.color
```

```gdscript
func _on_card_hover_enter(card: PanelContainer) -> void:
    ## 状态机检查：已选中/已禁用/满员的卡片不响应hover
    var state: String = card.get_meta("state", "normal")
    if state == "selected" or state == "disabled" or state == "in_team":
        return
    if _selected_team.size() >= PartyAssembleSettings.MAX_TEAM_SIZE and state != "selected":
        return
    
    ## kill旧tween（参考card-framework memory-safe Tween清理）
    _kill_card_tween(card)
    
    ## 新hover tween：scale放大 + Y轴上浮 + 阴影扩散（参考UiCard hovered size/height）
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    
    ## Scale放大1.08x（参考UiCard hovered size）
    tween.tween_property(card, "scale", PartyAssembleSettings.HOVER_SCALE, PartyAssembleSettings.HOVER_DURATION)
    
    ## Y轴上浮12px（参考UiCard hovered height）
    tween.parallel().tween_property(card, "position:y", card.position.y + PartyAssembleSettings.HOVER_LIFT_Y, PartyAssembleSettings.HOVER_DURATION)
    
    ## Z-index提升（参考card-framework z-index management）
    tween.parallel().tween_property(card, "z_index", 10, 0.0)
    
    ## 样式切换：蓝色边框 + 扩散阴影
    tween.parallel().tween_callback(func():
        card.add_theme_stylebox_override("panel", _create_card_style("hover"))
    )
    
    card.set_meta("hover_tween", tween)

## === Hover退出 ===
func _on_card_hover_exit(card: PanelContainer) -> void:
    var state: String = card.get_meta("state", "normal")
    if state == "selected":
        return
    
    _kill_card_tween(card)
    
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    tween.tween_property(card, "scale", Vector2.ONE, PartyAssembleSettings.HOVER_DURATION)
    tween.parallel().tween_property(card, "position:y", card.position.y - PartyAssembleSettings.HOVER_LIFT_Y, PartyAssembleSettings.HOVER_DURATION)
    tween.parallel().tween_property(card, "z_index", 0, 0.0)
    tween.parallel().tween_callback(func():
        if state != "disabled" and state != "in_team":
            card.add_theme_stylebox_override("panel", _create_card_style("normal"))
    )
    
    card.set_meta("hover_tween", tween)

## === 点击选择（加入队伍）===
func _on_card_gui_input(event: InputEvent, card: PanelContainer, partner: Dictionary) -> void:
    if not (event is InputEventMouseButton and event.pressed and event.button_index == MOUSE_BUTTON_LEFT):
        return
    
    var state: String = card.get_meta("state", "normal")
    if state == "disabled" or state == "in_team":
        AudioManager.play_ui("error")
        return
    if _selected_team.size() >= PartyAssembleSettings.MAX_TEAM_SIZE:
        AudioManager.play_ui("error")
        return
    
    AudioManager.play_ui("confirm")
    _add_to_team(partner, card)

## === 安全kill tween（参考card-framework memory management）===
func _kill_card_tween(card: PanelContainer) -> void:
    if card.has_meta("hover_tween"):
        var old: Tween = card.get_meta("hover_tween")
        if old != null and old.is_valid():
            old.kill()
        card.remove_meta("hover_tween")
```

```gdscript
func _add_to_team(partner: Dictionary, source_card: PanelContainer) -> void:
    if _selected_team.size() >= PartyAssembleSettings.MAX_TEAM_SIZE:
        return
    
    _selected_team.append(partner)
    
    ## 1. 标记池子卡片为"已入队"（禁用态，参考UiCard disabled transparency）
    source_card.set_meta("state", "in_team")
    _kill_card_tween(source_card)
    
    var fade_tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    fade_tween.tween_property(source_card, "modulate", Color(0.6, 0.6, 0.62, 0.4), 0.2)  ## 灰度+半透明
    fade_tween.parallel().tween_property(source_card, "scale", Vector2.ONE, 0.2)
    fade_tween.parallel().tween_property(source_card, "position:y", source_card.position.y - PartyAssembleSettings.HOVER_LIFT_Y, 0.2)
    
    ## 2. 更新队伍槽位（飞入效果）
    var slot_index: int = _selected_team.size() - 1
    _update_team_slot_with_fly_in(slot_index, partner, source_card.global_position)
    
    ## 3. 更新UI
    _update_team_title()
    _update_team_stats()
    
    if _selected_team.size() >= PartyAssembleSettings.MAX_TEAM_SIZE:
        _start_btn.disabled = false

## === 槽位飞入动画（卡片从池子位置飞到槽位）===
func _update_team_slot_with_fly_in(index: int, partner: Dictionary, from_global_pos: Vector2) -> void:
    var slot: PanelContainer = get_node("UILayer/MidSection/RightPanel/TeamSlots/TeamSlot_%d" % index)
    
    ## 清除空槽内容
    for child in slot.get_children():
        child.queue_free()
    
    ## 设置选中样式
    slot.add_theme_stylebox_override("panel", _create_card_style("selected"))
    
    ## 构建槽位内容（和池子卡片类似但稍小）
    var vbox := _build_slot_content(partner, slot)
    slot.add_child(vbox)
    
    ## 飞入动画：从池子卡片位置飞到槽位位置
    var slot_global := slot.global_position
    var fly_offset := from_global_pos - slot_global
    
    vbox.global_position = from_global_pos
    vbox.modulate.a = 0.0
    vbox.scale = Vector2(0.7, 0.7)
    
    var fly_tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    
    ## 先快速飞到目标位置（0.25s）
    fly_tween.tween_property(vbox, "global_position", slot_global, 0.25)
    fly_tween.parallel().tween_property(vbox, "modulate:a", 1.0, 0.2)
    fly_tween.parallel().tween_property(vbox, "scale", Vector2.ONE, 0.25)
    
    ## 到达后弹跳（参考card-framework状态机弹跳）
    fly_tween.chain().tween_property(vbox, "scale", Vector2(1.04, 1.04), 0.1).set_trans(Tween.TRANS_BACK)
    fly_tween.tween_property(vbox, "scale", Vector2.ONE, 0.15).set_trans(Tween.TRANS_BACK)
```

### 9. 入场动画（三卡片同时弹出 → 其余依次淡入）

```gdscript
func _play_entrance_animation() -> void:
    ## Step 1: 面板容器依次淡入+上滑
    var panels: Array[Control] = [
        $UILayer/TopBar,
        $UILayer/SelectedHeroBanner,
        $UILayer/MidSection/LeftPanel,
        $UILayer/MidSection/RightPanel,
        $UILayer/BottomBar,
    ]
    
    for panel in panels:
        panel.modulate.a = 0.0
        panel.position.y += 20
    
    var panel_tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    for i in range(panels.size()):
        var panel: Control = panels[i]
        var delay := i * 0.08
        panel_tween.tween_property(panel, "modulate:a", 1.0, PartyAssembleSettings.ENTRANCE_DURATION).set_delay(delay)
        panel_tween.parallel().tween_property(panel, "position:y", panel.position.y - 20, PartyAssembleSettings.ENTRANCE_DURATION + 0.05).set_delay(delay)
    
    ## Step 2: 池子卡片同时弹出（参考card-framework入场）
    await panel_tween.finished
    var cards := $UILayer/MidSection/LeftPanel/PartnerPool/PartnerPoolGrid.get_children()
    
    for card in cards:
        card.scale = Vector2(0.85, 0.85)
        card.modulate.a = 0.0
    
    ## 所有卡片同时开始，但每个delay递增0.05s（波浪效果）
    for i in range(cards.size()):
        var card: Control = cards[i]
        var card_tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT).set_delay(i * 0.05)
        card_tween.tween_property(card, "scale", Vector2.ONE, 0.3)
        card_tween.parallel().tween_property(card, "modulate:a", 1.0, 0.25)
    
    ## Step 3: 空槽位弹跳入场
    var slots := $UILayer/MidSection/RightPanel/TeamSlots.get_children()
    for i in range(slots.size()):
        var slot_tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT).set_delay(0.3 + i * 0.1)
        slot_tween.tween_property(slots[i], "scale", Vector2(1.02, 1.02), 0.2)
        slot_tween.tween_property(slots[i], "scale", Vector2.ONE, 0.15)
```
