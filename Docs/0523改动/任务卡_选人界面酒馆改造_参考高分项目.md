# 任务卡：选人界面（酒馆）改造（参考高分项目实现）

## 参考来源

1. **IYanel-DEV/MainMenu-MP** — 高星主菜单主题集合（已下载分析）
   - StyleBoxFlat + shadow_size/shadow_offset 制造层次感
   - White主题：纯白底+浅灰边框+蓝色hover边框

2. **Maaack/Godot-Menus-Template** — Godot官方推荐菜单模板（已下载分析）
   - AnimationTree入场动画方案
   - 子菜单封装模式 `_open_sub_menu()` / `_close_sub_menu()`

3. **matyo91/darkwaar2** — 炉石风卡牌游戏（420 stars）
   - `CardChoiceScene.tscn` 卡牌选择界面结构
   - Start → Card Selection → Game 流程

4. **guladam/deck_builder_tutorial** — Roguelike卡牌构建教程（420 stars）
   - 角色/卡牌选择UI布局
   - 左侧列表 + 右侧详情面板结构

5. **linyangqi/godot-card-game-framework-gd4** — 卡牌游戏框架
   - Card Library 列表式UI
   - 自动聚焦、hover放大、拖拽选择交互

```gdscript
func _create_hero_card(hero_id: String, config: Dictionary, is_unlocked: bool) -> PanelContainer:
    var card := PanelContainer.new()
    card.custom_minimum_size = Vector2(280, 80)
    card.size_flags_horizontal = Control.SIZE_FILL
    card.mouse_filter = Control.MOUSE_FILTER_STOP
    card.set_meta("hero_id", hero_id)
    card.set_meta("is_unlocked", is_unlocked)
    
    ## === 卡片样式：白底 + 浅灰边框 + 底部加粗（贴纸厚度感，参考MainMenu-MP）===

    style.border_color = COLOR_BORDER
    style.border_width_left = 2
    style.border_width_top = 2

    style.border_width_bottom = 3     ## 底部加粗 = 贴纸厚度

    style.corner_radius_bottom_left = 8
    style.corner_radius_bottom_right = 8

    style.shadow_offset = Vector2(0, 2)

    var hbox := HBoxContainer.new()
    hbox.alignment = BoxContainer.ALIGNMENT_BEGIN

    if not is_unlocked:

        var lock_icon := TextureRect.new()

        hbox.add_child(lock_icon)
    
    ## === 交互信号 ===
    card.gui_input.connect(func(event: InputEvent):
        if event is InputEventMouseButton and event.pressed and event.button_index == MOUSE_BUTTON_LEFT:
            _select_hero(hero_id, card, is_unlocked)

    card.mouse_entered.connect(func(): _card_hover_tween(card, true))
    card.mouse_exited.connect(func(): _card_hover_tween(card, false))
    
    return card
```

### 3. 卡片 Hover 动画（参考 card-game-framework 自动聚焦）

```gdscript
func _card_hover_tween(card: PanelContainer, is_enter: bool) -> void:
    if card == _selected_card:
        return
    if card.has_meta("hover_tween"):
        var old: Tween = card.get_meta("hover_tween")
        if old != null and old.is_valid():
            old.kill()
    
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    if is_enter:
        ## hover：微缩放 + 蓝色边框 + 阴影扩散（MainMenu-MP标准hover效果）
        tween.tween_property(card, "scale", Vector2(1.02, 1.02), 0.12)
        
        var hover_style: StyleBoxFlat = card.get_theme_stylebox("panel").duplicate()
        hover_style.border_color = COLOR_BORDER_HOVER    ## 变蓝
        hover_style.shadow_size = 10                      ## 阴影扩散
        hover_style.shadow_color = Color(0.4, 0.6, 1.0, 0.15)
        card.add_theme_stylebox_override("panel", hover_style)
    else:
        ## 恢复
        tween.tween_property(card, "scale", Vector2.ONE, 0.15)
        
        var base_style: StyleBoxFlat = card.get_theme_stylebox("panel").duplicate()
        base_style.border_color = COLOR_BORDER
        base_style.shadow_size = 6
        base_style.shadow_color = Color(0, 0, 0, 0.08)
        card.add_theme_stylebox_override("panel", base_style)
    
    card.set_meta("hover_tween", tween)
```

### 4. 选中态（左侧指示条 + 阴影扩散）

```gdscript
func _select_hero(hero_id: String, card: PanelContainer, is_unlocked: bool) -> void:
    _selected_hero_id = hero_id
    
    ## 取消旧选中
    if _selected_card != null and is_instance_valid(_selected_card):
        var old_style: StyleBoxFlat = _selected_card.get_theme_stylebox("panel").duplicate()
        old_style.bg_color = COLOR_BG_PANEL
        old_style.border_color = COLOR_BORDER
        old_style.border_width_left = 2
        old_style.shadow_size = 6
        _selected_card.add_theme_stylebox_override("panel", old_style)
        _selected_card.modulate = Color.WHITE
        
        var unselect_tween := create_tween()
        unselect_tween.tween_property(_selected_card, "scale", Vector2.ONE, 0.15)
    
    _selected_card = card
    
    ## 新选中态：微蓝白底 + 左侧蓝色粗边框（选中指示）+ 阴影扩散
    var new_style: StyleBoxFlat = card.get_theme_stylebox("panel").duplicate()
    new_style.bg_color = COLOR_BG_SELECTED
    new_style.border_color = COLOR_BORDER_HOVER
    new_style.border_width_left = 4          ## 左侧4px蓝色 = 选中指示
    new_style.border_width_top = 2
    new_style.border_width_right = 2
    new_style.border_width_bottom = 3
    new_style.shadow_size = 12               ## 选中时阴影最重
    new_style.shadow_color = Color(0.4, 0.6, 1.0, 0.2)
    card.add_theme_stylebox_override("panel", new_style)
    card.modulate = Color.WHITE
    
    ## 选中入场动画：scale弹跳（参考Maaack动画风格）
    var select_tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT)
    select_tween.tween_property(card, "scale", Vector2(1.03, 1.03), 0.15)
    select_tween.tween_property(card, "scale", Vector2.ONE, 0.2)
    
    _update_right_panel(hero_id, is_unlocked)
    _update_start_button(is_unlocked)
```

### 6. 立绘切换动画（淡入淡出）

```gdscript
func _update_right_panel(hero_id: String, is_unlocked: bool) -> void:
    var config: Dictionary = ConfigManager.get_hero_config(hero_id)
    if config.is_empty():
        return
    
    var portrait_path: String = ConfigManager.get_hero_portrait_path(hero_id)
    
    ## 1. 当前立绘淡出
    var tween_out := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    tween_out.tween_property(_big_portrait, "modulate:a", 0.0, 0.12)
    await tween_out.finished
    
    if is_unlocked:
        ## 加载立绘
        if not portrait_path.is_empty():
            var tex: Texture2D = load(portrait_path)
            if tex != null:
                _big_portrait.texture = tex
                _big_portrait.modulate = Color.WHITE
            else:
                _big_portrait.texture = null
                _big_portrait.modulate = Color.html(config.get("portrait_color", "#888888"))
        _silhouette_label.visible = false
        _hero_name_label.text = config.get("hero_name", "")
        _hero_role_label.text = config.get("class_desc", "")
        _hero_name_label.add_theme_color_override("font_color", COLOR_TEXT_MAIN)
        _hero_role_label.add_theme_color_override("font_color", COLOR_TEXT_SECOND)
        _lock_overlay.visible = false
    else:
        ## 未解锁：灰色占位
        _big_portrait.texture = null
        _big_portrait.modulate = Color(0.85, 0.85, 0.87, 1)
        _silhouette_label.visible = true
        _silhouette_label.add_theme_color_override("font_color", COLOR_TEXT_SECOND)
        _hero_name_label.text = "???"
        _hero_role_label.text = "???"
        _hero_name_label.add_theme_color_override("font_color", COLOR_TEXT_SECOND)
        _hero_role_label.add_theme_color_override("font_color", COLOR_TEXT_SECOND)
        _lock_overlay.visible = true
        _lock_overlay.color = Color(1.0, 1.0, 1.0, 0.6)
        var condition: String = config.get("unlock_condition_text", "")
        _lock_label.text = condition if not condition.is_empty() else "未解锁"
        _lock_label.add_theme_color_override("font_color", COLOR_TEXT_GOLD)
    
    ## 2. 新立绘淡入 + scale弹跳
    _big_portrait.scale = Vector2(0.98, 0.98)
    var tween_in := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    tween_in.tween_property(_big_portrait, "modulate:a", 1.0, 0.18)
    tween_in.parallel().tween_property(_big_portrait, "scale", Vector2.ONE, 0.25).set_trans(Tween.TRANS_BACK)
```

```gdscript
    _set_toggle_state(_novice_btn, not _is_veteran_mode)

func _set_toggle_state(btn: Button, is_active: bool) -> void:
    var style := StyleBoxFlat.new()

        style.bg_color = COLOR_BORDER_HOVER      ## 活力橙/蓝色底

    else:
        style.bg_color = Color(0, 0, 0, 0)       ## 透明
        btn.add_theme_color_override("font_color", COLOR_TEXT_SECOND)
    btn.add_theme_stylebox_override("normal", style)
    btn.add_theme_stylebox_override("pressed", style)
    btn.add_theme_stylebox_override("hover", style)
```

### 10. 入场动画（参考 Maaack AnimationTree 方案，用Tween简化实现）

```gdscript
func _ready() -> void:

func _play_entrance_animation() -> void:
    ## 初始态：所有面板下移+透明
    var panels: Array[Control] = [
        $UILayer/TopBar,
        $UILayer/LeftPanel,
        $UILayer/RightPanel,
        $UILayer/BottomPanel,
    ]
    
    for panel in panels:
        panel.modulate.a = 0.0
        panel.position.y += 25
    
    ## 顺序入场：顶部栏 → 左侧列表 → 右侧立绘 → 底部属性
    var seq_tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    
    for i in range(panels.size()):
        var panel: Control = panels[i]
        var delay: float = i * 0.1
        seq_tween.tween_property(panel, "modulate:a", 1.0, 0.35).set_delay(delay)
        seq_tween.parallel().tween_property(panel, "position:y", panel.position.y - 25, 0.4).set_delay(delay)
    
    ## 立绘特殊：额外scale弹跳
    await seq_tween.finished
    var portrait_tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT)
    portrait_tween.tween_property(_big_portrait, "scale", Vector2(1.02, 1.02), 0.3)
    portrait_tween.tween_property(_big_portrait, "scale", Vector2.ONE, 0.25)
```
