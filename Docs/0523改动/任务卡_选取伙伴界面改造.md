## 界面定位

**触发时机**：冒险过程中获得招募伙伴机会时弹出（如：完成营救、商店招募、击败BOSS奖励、特殊事件）

**形式**：全屏弹窗（PanelContainer 居中，带半透明背景遮罩），支持 2~4 个伙伴可选

## Step 2：伙伴卡片（核心交互元素）

### 卡片尺寸与布局

```gdscript
# 卡片常量
const CARD_SIZE := Vector2(240, 320)      # 竖版卡片，类似炉石/杀戮尖塔
const AVATAR_SIZE := Vector2(160, 160)     # 正方形头像框（设计文档确认）

func _create_partner_card(partner_data: Dictionary, index: int) -> PanelContainer:
    var card := PanelContainer.new()
    card.custom_minimum_size = CARD_SIZE
    card.size_flags_horizontal = Control.SIZE_SHRINK_CENTER
    card.set_meta("partner_id", partner_data.get("id", ""))
    card.set_meta("partner_data", partner_data)
    card.set_meta("card_index", index)
    
    # === 卡片基础样式 ===
    var style := StyleBoxFlat.new()
    style.bg_color = Color(1.0, 1.0, 1.0, 1.0)
    style.border_color = Color(0.85, 0.85, 0.85, 1.0)
    style.border_width_left = 2
    style.border_width_top = 2
    style.border_width_right = 2
    style.border_width_bottom = 3             # 底部加粗=贴纸厚度
    style.corner_radius_top_left = 12
    style.corner_radius_top_right = 12
    style.corner_radius_bottom_left = 12
    style.corner_radius_bottom_right = 12
    style.shadow_color = Color(0, 0, 0, 0.08)
    style.shadow_size = 6
    style.shadow_offset = Vector2(0, 2)
    card.add_theme_stylebox_override("panel", style)
    
    # === 内部垂直布局 ===
    var vbox := VBoxContainer.new()
    vbox.alignment = BoxContainer.ALIGNMENT_CENTER
    vbox.add_theme_constant_override("separation", 8)
    card.add_child(vbox)
    
    # === 等级标签（顶部角标）===
    var level_badge := Label.new()
    level_badge.text = "Lv.%d" % partner_data.get("level", 1)
    level_badge.add_theme_font_size_override("font_size", 12)
    level_badge.add_theme_color_override("font_color", Color(0.85, 0.65, 0.15, 1))  # 亮金
    vbox.add_child(level_badge)
    
    # === 正方形头像框 ===
    var avatar_container := PanelContainer.new()
    avatar_container.custom_minimum_size = AVATAR_SIZE
    var avatar_style := StyleBoxFlat.new()
    avatar_style.bg_color = Color(0.95, 0.95, 0.97, 1)
    avatar_style.corner_radius_top_left = 8
    avatar_style.corner_radius_top_right = 8
    avatar_style.corner_radius_bottom_left = 8
    avatar_style.corner_radius_bottom_right = 8
    avatar_container.add_theme_stylebox_override("panel", avatar_style)
    vbox.add_child(avatar_container)
    
    var avatar := TextureRect.new()
    avatar.expand_mode = TextureRect.EXPAND_FIT_WIDTH_PROPORTIONAL
    avatar.stretch_mode = TextureRect.STRETCH_KEEP_ASPECT_CENTERED
    avatar.custom_minimum_size = AVATAR_SIZE
    var avatar_path: String = partner_data.get("avatar_path", "")
    if not avatar_path.is_empty():
        var tex: Texture2D = load(avatar_path)
        if tex != null:
            avatar.texture = tex
    avatar_container.add_child(avatar)
    
    # === 伙伴名字 ===
    var name_label := Label.new()
    name_label.text = partner_data.get("name", "???")
    name_label.add_theme_font_size_override("font_size", 18)
    name_label.add_theme_color_override("font_color", Color(0.2, 0.2, 0.2, 1))
    name_label.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
    vbox.add_child(name_label)
    
    # === 定位/职业 ===
    var class_label := Label.new()
    class_label.text = partner_data.get("class_desc", "伙伴")
    class_label.add_theme_font_size_override("font_size", 12)
    class_label.add_theme_color_override("font_color", Color(0.5, 0.5, 0.5, 1))
    class_label.horizontal_alignment = HORIZONTAL_ALIGNMENT_CENTER
    vbox.add_child(class_label)
    
    # === 稀有度装饰条 ===
    var rarity_bar := ColorRect.new()
    rarity_bar.custom_minimum_size = Vector2(200, 4)
    rarity_bar.color = _get_rarity_color(partner_data.get("rarity", "common"))
    vbox.add_child(rarity_bar)
    
    # === 交互信号 ===
    card.gui_input.connect(func(event: InputEvent):
        if event is InputEventMouseButton and event.pressed and event.button_index == MOUSE_BUTTON_LEFT:
            _select_partner_card(card, partner_data)
    )
    card.mouse_entered.connect(func(): _card_hover_tween(card, true))
    card.mouse_exited.connect(func(): _card_hover_tween(card, false))
    
    return card

func _get_rarity_color(rarity: String) -> Color:
    match rarity:
        "legendary": return Color(0.9, 0.5, 0.1, 1)   # 橙
        "epic": return Color(0.7, 0.3, 0.9, 1)        # 紫
        "rare": return Color(0.2, 0.5, 0.9, 1)        # 蓝
        _: return Color(0.6, 0.6, 0.6, 1)             # 灰
```

### 卡片 Hover 效果

```gdscript
func _card_hover_tween(card: PanelContainer, is_enter: bool) -> void:
    if card == _selected_card:
        return
    if card.has_meta("hover_tween"):
        var old: Tween = card.get_meta("hover_tween")
        if old != null and old.is_valid():
            old.kill()
    
    var tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_OUT)
    if is_enter:
        # hover：上浮+阴影扩散+边框变蓝
        tween.tween_property(card, "scale", Vector2(1.05, 1.05), 0.15)
        tween.parallel().tween_property(card, "position:y", card.position.y - 8, 0.15)
        
        var hover_style: StyleBoxFlat = card.get_theme_stylebox("panel").duplicate()
        hover_style.border_color = Color(0.4, 0.6, 1.0, 1)   # 蓝边框
        hover_style.shadow_size = 12
        hover_style.shadow_color = Color(0.4, 0.6, 1.0, 0.2)
        card.add_theme_stylebox_override("panel", hover_style)
    else:
        tween.tween_property(card, "scale", Vector2.ONE, 0.2)
        tween.parallel().tween_property(card, "position:y", card.position.y + 8, 0.2)
        
        var base_style: StyleBoxFlat = card.get_theme_stylebox("panel").duplicate()
        base_style.border_color = Color(0.85, 0.85, 0.85, 1.0)
        base_style.shadow_size = 6
        base_style.shadow_color = Color(0, 0, 0, 0.08)
        card.add_theme_stylebox_override("panel", base_style)
    
    card.set_meta("hover_tween", tween)
```

var _selected_card: PanelContainer = null
var _selected_partner_data: Dictionary = {}

func _select_partner_card(card: PanelContainer, partner_data: Dictionary) -> void:
    _selected_partner_data = partner_data
    
    # 取消旧选中
    if _selected_card != null and is_instance_valid(_selected_card):
        var old_style: StyleBoxFlat = _selected_card.get_theme_stylebox("panel").duplicate()
        old_style.bg_color = Color(1.0, 1.0, 1.0, 1.0)
        old_style.border_color = Color(0.85, 0.85, 0.85, 1.0)
        old_style.border_width_left = 2
        old_style.shadow_size = 6
        _selected_card.add_theme_stylebox_override("panel", old_style)
        _selected_card.modulate = Color.WHITE
        
        var unselect_tween := create_tween()
        unselect_tween.tween_property(_selected_card, "scale", Vector2.ONE, 0.15)
    
    _selected_card = card
    
    # 新选中态：蓝底 + 蓝色边框 + 阴影扩散
    var new_style: StyleBoxFlat = card.get_theme_stylebox("panel").duplicate()
    new_style.bg_color = Color(0.96, 0.98, 1.0, 1.0)       # 微蓝白底
    new_style.border_color = Color(0.4, 0.6, 1.0, 1.0)      # 蓝边框
    new_style.border_width_left = 3
    new_style.border_width_top = 3
    new_style.border_width_right = 3
    new_style.border_width_bottom = 4
    new_style.shadow_size = 16
    new_style.shadow_color = Color(0.4, 0.6, 1.0, 0.25)
    card.add_theme_stylebox_override("panel", new_style)
    
    # 选中弹跳动画
    var select_tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT)
    select_tween.tween_property(card, "scale", Vector2(1.06, 1.06), 0.12)
    select_tween.tween_property(card, "scale", Vector2(1.05, 1.05), 0.18)
    
    # 更新详情面板
    _update_info_panel(partner_data)
    
    # 启用确认按钮
    _confirm_btn.disabled = false
    _confirm_btn.modulate = Color.WHITE

## Step 5：入场/出场动画

```gdscript
func show_popup(partner_options: Array[Dictionary]) -> void:
    visible = true
    _build_partner_cards(partner_options)
    
    # 遮罩淡入
    var dim: ColorRect = $DimOverlay
    dim.color = Color(0, 0, 0, 0)
    var dim_tween := create_tween()
    dim_tween.tween_property(dim, "color:a", 0.5, 0.3)
    
    # 主面板缩放入场
    var panel: PanelContainer = $MainPanel
    panel.scale = Vector2(0.9, 0.9)
    panel.modulate.a = 0.0
    
    var panel_tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT)
    panel_tween.tween_property(panel, "scale", Vector2.ONE, 0.35)
    panel_tween.parallel().tween_property(panel, "modulate:a", 1.0, 0.3)
    
    # 卡片依次入场
    await panel_tween.finished
    var cards := $MainPanel/VBoxContainer/PartnerCardsContainer.get_children()
    for i in range(cards.size()):
        var card: Control = cards[i]
        card.scale = Vector2(0.8, 0.8)
        card.modulate.a = 0.0
        
        var card_tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT).set_delay(i * 0.08)
        card_tween.tween_property(card, "scale", Vector2.ONE, 0.25)
        card_tween.parallel().tween_property(card, "modulate:a", 1.0, 0.2)

func hide_popup() -> void:
    # 主面板淡出缩小
    var panel: PanelContainer = $MainPanel
    var panel_tween := create_tween().set_trans(Tween.TRANS_QUAD).set_ease(Tween.EASE_IN)
    panel_tween.tween_property(panel, "scale", Vector2(0.9, 0.9), 0.2)
    panel_tween.parallel().tween_property(panel, "modulate:a", 0.0, 0.2)
    
    # 遮罩淡出
    var dim: ColorRect = $DimOverlay
    var dim_tween := create_tween()
    dim_tween.tween_property(dim, "color:a", 0.0, 0.25)
    
    await panel_tween.finished
    visible = false
    _selected_card = null
    _selected_partner_data = {}
```

    if _selected_partner_data.is_empty():
        return

    var tween := create_tween().set_trans(Tween.TRANS_BACK).set_ease(Tween.EASE_OUT)

        _selected_partner_data
    )
    hide_popup()

    AudioManager.play_ui("cancel")
    popup_cancelled.emit()
    hide_popup()

# 在获得伙伴招募机会时调用
func _show_partner_selection(partner_options: Array[Dictionary]) -> void:
    var popup: PartnerSelectPopup = $UILayer/PartnerSelectPopup
    popup.partner_selected.connect(_on_partner_selected, CONNECT_ONE_SHOT)
    popup.popup_cancelled.connect(_on_partner_cancelled, CONNECT_ONE_SHOT)
    popup.show_popup(partner_options)

func _on_partner_selected(partner_id: String, partner_data: Dictionary) -> void:
    print("[Partner] 招募伙伴: %s" % partner_id)
    # 添加到队伍
    GameManager.add_partner(partner_id, partner_data)
    # 更新爬塔UI
    _update_party_display()

func _on_partner_cancelled() -> void:
    print("[Partner] 取消招募")
