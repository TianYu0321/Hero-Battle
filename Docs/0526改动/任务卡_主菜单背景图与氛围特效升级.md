## Step 2：Parallax2D多层背景设置

```gdscript
## menu.gd — 视差背景初始化

@onready var parallax_bg: ParallaxBackground = $ParallaxBackground

func _setup_parallax_background() -> void:
    ## 第0层：天空（几乎不动）
    var layer0: Parallax2D = $ParallaxBackground/Layer0_Sky
    layer0.scroll_scale = Vector2(0.05, 0)
    layer0.repeat_size = Vector2(2560, 0)  # 水平无缝循环
    _setup_layer_texture(layer0, ResourcePaths.get_menu_bg_layer(0))
    
    ## 第1层：山脉（缓慢）
    var layer1: Parallax2D = $ParallaxBackground/Layer1_Mountains
    layer1.scroll_scale = Vector2(0.15, 0)
    layer1.repeat_size = Vector2(2560, 0)
    _setup_layer_texture(layer1, ResourcePaths.get_menu_bg_layer(1))
    
    ## 第2层：丘陵+村庄（中速）
    var layer2: Parallax2D = $ParallaxBackground/Layer2_Hills
    layer2.scroll_scale = Vector2(0.35, 0)
    layer2.repeat_size = Vector2(2560, 0)
    _setup_layer_texture(layer2, ResourcePaths.get_menu_bg_layer(2))
    
    ## 第3层：树林（较快）
    var layer3: Parallax2D = $ParallaxBackground/Layer3_Trees
    layer3.scroll_scale = Vector2(0.6, 0)
    layer3.repeat_size = Vector2(2560, 0)
    _setup_layer_texture(layer3, ResourcePaths.get_menu_bg_layer(3))
    
    ## 第4层：草地（正常速度，作为前景底部）
    var layer4: Parallax2D = $ParallaxBackground/Layer4_Ground
    layer4.scroll_scale = Vector2(1.0, 0)
    ## 前景不循环，固定底部
    _setup_layer_texture(layer4, ResourcePaths.get_menu_bg_layer(4), false)
    
    ## 初始偏移：让画面居中展示最佳构图
    parallax_bg.scroll_offset = Vector2(200, 0)

func _setup_layer_texture(layer: Parallax2D, texture_path: String, enable_repeat: bool = true) -> void:
    var sprite: Sprite2D = layer.get_node("Sprite2D")
    
    var tex: Texture2D = ResourcePaths.load_texture_safe(texture_path)
    if tex == null:
        push_warning("Menu background layer texture not found: %s" % texture_path)
        return
    
    sprite.texture = tex
    sprite.centered = false  # 左上角对齐(0,0)，视差要求
    sprite.position = Vector2.ZERO
    
    if enable_repeat:
        ## 确保纹理重复模式正确
        tex.repeat = 1  # CanvasItem.TEXTURE_REPEAT_ENABLED
        layer.repeat_size = Vector2(tex.get_width(), 0)
    
    ## 垂直居中或底部对齐
    var viewport_h := get_viewport().get_visible_rect().size.y
    var tex_h := tex.get_height()
    
    ## 如果纹理高度不足1080，垂直拉伸或居中
    if tex_h < viewport_h:
        sprite.scale.y = viewport_h / tex_h
    
    ## 水平铺满：如果纹理宽度不够，scale.x适配
    var viewport_w := get_viewport().get_visible_rect().size.x
    var tex_w := tex.get_width()
    if tex_w < viewport_w and not enable_repeat:
        sprite.scale.x = viewport_w / tex_w

## ========== 动态视差效果（鼠标/自动漂移）==========

var _parallax_auto_scroll: float = 0.0
const PARALLAX_AUTO_SPEED: float = 5.0  # 像素/秒，缓慢自动漂移

func _process(delta: float) -> void:
    ## 自动缓慢漂移（营造微风感）
    _parallax_auto_scroll += PARALLAX_AUTO_SPEED * delta
    parallax_bg.scroll_offset.x = _parallax_auto_scroll
    
    ## 鼠标视差（可选：鼠标位置影响偏移）
```

### 备选：不用ParallaxBackground，用TextureRect多层（更简单）

如果Godot 4.x版本对Parallax2D支持有问题，使用简单多层TextureRect方案：

```gdscript
## 简单多层背景（无true视差，用代码模拟）

@onready var bg_layers: Array[TextureRect] = [
    $BackgroundLayer/Layer0_Sky,
    $BackgroundLayer/Layer1_Mountains,
    $BackgroundLayer/Layer2_Hills,
    $BackgroundLayer/Layer3_Trees,
    $BackgroundLayer/Layer4_Ground,
]

const LAYER_SPEEDS := [0.05, 0.15, 0.35, 0.6, 1.0]

func _ready() -> void:
    _setup_simple_parallax()

func _setup_simple_parallax() -> void:
    for i in range(bg_layers.size()):
        var layer: TextureRect = bg_layers[i]
        layer.set_anchors_preset(Control.PRESET_FULL_RECT)
        layer.expand_mode = TextureRect.EXPAND_IGNORE_SIZE
        layer.stretch_mode = TextureRect.STRETCH_KEEP_ASPECT_CENTERED
        
        ## 确保铺满：放大到至少覆盖屏幕
        var tex: Texture2D = layer.texture
        if tex != null:
            var viewport_size := get_viewport().get_visible_rect().size
            var scale_needed := maxf(
                viewport_size.x / tex.get_width(),
                viewport_size.y / tex.get_height()
            )
            layer.scale = Vector2(scale_needed, scale_needed)

func _process(delta: float) -> void:
    ## 各层以不同速度水平移动
    for i in range(bg_layers.size()):
        var layer: TextureRect = bg_layers[i]
        var speed: float = LAYER_SPEEDS[i] * PARALLAX_AUTO_SPEED
        layer.position.x -= speed * delta
        
        ## 循环复位
        var tex_w: float = layer.texture.get_width() * layer.scale.x if layer.texture else 1920
        if layer.position.x <= -tex_w:
            layer.position.x += tex_w
```

```gdscript
func _setup_fireflies() -> void:
    var particles: CPUParticles2D = $AmbientParticles/Fireflies
    
    ## 发射器设置
    particles.emitting = true
    particles.amount = 30
    particles.lifetime = 4.0
    particles.preprocess = 4.0  # 预运行，避免开始时稀疏
    
    ## 发射区域：全屏
    particles.emission_shape = CPUParticles2D.EMISSION_SHAPE_RECTANGLE
    particles.emission_rect_extents = Vector2(960, 540)  # 半屏范围，中心发射
    particles.position = Vector2(960, 540)
    
    ## 物理
    particles.direction = Vector2(0, -1)
    particles.spread = 30.0
    particles.gravity = Vector2(0, -5)  # 轻微向上飘（营造轻盈感）
    particles.initial_velocity_min = 10.0
    particles.initial_velocity_max = 30.0
    
    ## 外观
    particles.scale_amount_min = 0.5
    particles.scale_amount_max = 1.5
    particles.scale_amount_curve = _create_pulse_curve()  # 脉动缩放
    
    ## 颜色：暖黄绿色（萤火虫）
    particles.color = Color(0.9, 1.0, 0.4, 0.8)
    particles.color_ramp = _create_firefly_color_ramp()
    
    ## 闪烁效果（通过alpha曲线）
    particles.alpha_curve = _create_blink_curve()
    
    ## 混合模式：加法发光
    particles.draw_order = CPUParticles2D.DRAW_ORDER_INDEX

func _create_firefly_color_ramp() -> Gradient:
    var grad := Gradient.new()
    grad.add_point(0.0, Color(0.9, 1.0, 0.4, 0.0))   # 开头透明
    grad.add_point(0.2, Color(0.9, 1.0, 0.4, 1.0))   # 快速亮
    grad.add_point(0.8, Color(0.9, 1.0, 0.4, 1.0))   # 保持
    grad.add_point(1.0, Color(0.9, 1.0, 0.4, 0.0))   # 结尾淡出
    return grad

func _create_blink_curve() -> Curve:
    ## 模拟萤火虫闪烁：亮-暗-亮-暗
    var curve := Curve.new()
    curve.add_point(Vector2(0.0, 1.0))
    curve.add_point(Vector2(0.25, 0.3))
    curve.add_point(Vector2(0.5, 1.0))
    curve.add_point(Vector2(0.75, 0.2))
    curve.add_point(Vector2(1.0, 0.0))
    return curve

func _create_pulse_curve() -> Curve:
    var curve := Curve.new()
    curve.add_point(Vector2(0.0, 0.5))
    curve.add_point(Vector2(0.5, 1.0))
    curve.add_point(Vector2(1.0, 0.5))
    return curve
```

```gdscript
func _create_circle_texture(size: int, color: Color) -> Texture2D:
    var img := Image.create(size, size, false, Image.FORMAT_RGBA8)
    img.fill(Color.TRANSPARENT)
    
    var center := Vector2(size / 2, size / 2)
    var radius := size / 2 - 1
    
    for x in range(size):
        for y in range(size):
            if Vector2(x, y).distance_to(center) <= radius:
                img.set_pixel(x, y, color)
    
    return ImageTexture.create_from_image(img)
```

```gdscript
func _setup_vignette() -> void:
    var vignette: TextureRect = $VignetteOverlay
    
    vignette.set_anchors_preset(Control.PRESET_FULL_RECT)
    vignette.expand_mode = TextureRect.EXPAND_IGNORE_SIZE
    vignette.stretch_mode = TextureRect.STRETCH_KEEP_ASPECT_CENTERED

    var vignette_tex := _create_vignette_texture(1920, 1080, 0.4)
    vignette.texture = vignette_tex
    vignette.modulate = Color(1, 1, 1, 0.3)  # 30%透明度

func _create_vignette_texture(width: int, height: int, strength: float) -> Texture2D:
    var img := Image.create(width, height, false, Image.FORMAT_RGBA8)
    var center := Vector2(width / 2, height / 2)
    var max_dist := center.length() * 0.8
    
    for x in range(width):
        for y in range(height):
            var dist := Vector2(x, y).distance_to(center)
            var t := clampf(dist / max_dist, 0.0, 1.0)
            var alpha := pow(t, 2.0) * strength
            img.set_pixel(x, y, Color(0, 0, 0, alpha))
    
    return ImageTexture.create_from_image(img)
```

## Step 6：背景图铺满问题解决

### 核心问题：TextureRect不铺满1920x1080

```gdscript
## 正确的TextureRect铺满设置：

func _setup_single_bg_texture(bg: TextureRect, texture_path: String) -> void:
    bg.set_anchors_preset(Control.PRESET_FULL_RECT)
    bg.expand_mode = TextureRect.EXPAND_IGNORE_SIZE
    
    ## 关键：stretch_mode决定铺满方式
    ## KEEP_ASPECT_CENTERED = 保持比例居中裁剪（推荐，不变形）
    ## KEEP_ASPECT_COVERED = 保持比例覆盖全屏（可能裁剪）
    ## SCALE = 拉伸变形铺满（不推荐）
    bg.stretch_mode = TextureRect.STRETCH_KEEP_ASPECT_CENTERED
    
    var tex: Texture2D = ResourcePaths.load_texture_safe(texture_path)
    if tex != null:
        bg.texture = tex
        
        ## 如果纹理尺寸远小于1920x1080，需要放大
        var tex_size := Vector2(tex.get_width(), tex.get_height())
        var viewport_size := get_viewport().get_visible_rect().size
        
        ## 检查纹理是否足够大
        if tex_size.x < viewport_size.x or tex_size.y < viewport_size.y:
            push_warning("Background texture too small (%dx%d) for viewport (%dx%d), will be upscaled" % [
                int(tex_size.x), int(tex_size.y), int(viewport_size.x), int(viewport_size.y)
```

### 推荐背景图规格

```
分辨率要求：
- 最低：1920x1080（正好铺满）
- 推荐：2560x1080（超宽，支持视差滚动）
- 格式：PNG（无损）或 JPG（压缩，适合大背景）

如果只有小图：
- 使用 texture_filter = CanvasItem.TEXTURE_FILTER_NEAREST_WITH_MIPMAPS（避免模糊）
- 或在TextureRect上设置 texture_filter
```

## 测试验收标准

### 背景图铺满
- [ ] 1920x1080分辨率下背景完全覆盖屏幕，无黑边
- [ ] 2560x1080超宽屏下背景也铺满（或居中裁剪）
- [ ] 1280x720窗口模式下背景自适应铺满
- [ ] 背景图不变形（保持宽高比）
- [ ] 背景图不模糊（texture_filter设置正确）

### 多层视差
- [ ] 5层背景（天空/山/丘陵/树林/草地）
- [ ] 每层以不同速度水平移动（0.05/0.15/0.35/0.6/1.0）
- [ ] 自动缓慢漂移（5px/秒）
- [ ] 无缝循环（repeat_size或代码复位）
- [ ] 鼠标移动可选影响视差偏移
- [ ] 各层texture左上角对齐(0,0)，无闪烁

### 氛围粒子
- [ ] 萤火虫：30个，暖黄绿色，脉动闪烁，缓慢上升飘落
- [ ] 尘埃光点：50个，暖白色，从顶部斜向飘落，微弱旋转
- [ ] 花瓣（可选）：15个，樱花粉色，旋转飘落，左右摇摆
- [ ] 粒子不遮挡UI（AmbientParticles在独立CanvasLayer）
- [ ] 粒子不超出屏幕边界（emission_rect设置正确）
- [ ] 启动时无粒子稀疏期（preprocess设置）

### 暗角遮罩
- [ ] 屏幕边缘有轻微暗角（30%透明度）
- [ ] 中间区域保持明亮（UI可读）
- [ ] 暗角不突兀（渐变柔和）

### 性能
- [ ] 5层视差+3组粒子+暗角，FPS保持在60（Godot 4兼容模式）
- [ ] 粒子数量可调（低配置减少amount）
- [ ] 背景纹理内存占用合理（JPG压缩大图）

### 资源管理
- [ ] 所有背景路径通过ResourcePaths.get_menu_bg_layer()加载
